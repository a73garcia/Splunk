def clear_screen() -> None:
    try:
        sys.stdout.write("\033[2J\033[3J\033[H")
        sys.stdout.flush()
    except Exception:
        os.system("cls")


def render_screen(render_fn) -> None:
    old_stdout = sys.stdout
    buffer = StringIO()
    sys.stdout = buffer
    try:
        render_fn()
    finally:
        sys.stdout = old_stdout

    clear_screen()
    print(buffer.getvalue(), end="", flush=True)


def strip_ansi(text: str) -> str:
    s = str(text)
    for code in ANSI_CODES:
        s = s.replace(code, "")
    return s


def ansi_center(text: str, width: int) -> str:
    plain = strip_ansi(text)
    if len(plain) >= width:
        return text
    left = (width - len(plain)) // 2
    right = width - len(plain) - left
    return (" " * left) + text + (" " * right)


def ansi_rjust(text: str, width: int) -> str:
    plain = strip_ansi(text)
    if len(plain) >= width:
        return text
    return (" " * (width - len(plain))) + text


def load_env_file(path: str) -> Dict[str, str]:
    data: Dict[str, str] = {}
    with open(path, "r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line or line.startswith("#") or "=" not in line:
                continue
            k, v = line.split("=", 1)
            data[k.strip()] = v.strip()
    return data

def fmt_num_dot(n: int) -> str:

def to_int(value: Optional[str]) -> int:
    if value is None:
        return 0

    s = str(value).strip()
    if not s or s.upper() == "N/A":
        return 0

    digits = []
    for ch in s:
        if ch.isdigit() or ch == ".":
            digits.append(ch)
        else:
            break

    try:
        return int(float("".join(digits))) if digits else 0
    except Exception:
        return 0

def terminal_width(default: int = 180) -> int:
    try:
        return os.get_terminal_size().columns

def visible_text_len(text: str) -> int:
    return len(strip_ansi(str(text)))


def pad_or_trim(text: str, width: int) -> str:
    text = str(text)
    plain = strip_ansi(text)
    if len(plain) <= width:
        return text + (" " * (width - len(plain)))
    if width <= 3:
        return plain[:width]
    return plain[: width - 3] + "..."


def block_max_width(lines: List[str], min_width: int = 20) -> int:
    if not lines:
        return min_width
    return max(min_width, max(visible_text_len(line) for line in lines))


def choose_three_col_widths(
    lines1: List[str],
    lines2: List[str],
    lines3: List[str],
    min1: int = 42,
    min2: int = 34,
    min3: int = 34,
    gap: int = 3,
) -> Tuple[int, int, int]:
    term_w = terminal_width()
    gaps_total = gap * 2
    available = max(60, term_w - gaps_total)

    need1 = block_max_width(lines1, min1)
    need2 = block_max_width(lines2, min2)
    need3 = block_max_width(lines3, min3)

    total_need = need1 + need2 + need3
    if total_need <= available:
        return need1, need2, need3

    weights = [need1, need2, need3]
    mins = [min1, min2, min3]

    extra_available = available - sum(mins)
    if extra_available <= 0:
        return tuple(mins)

    extra_needs = [max(0, w - m) for w, m in zip(weights, mins)]
    total_extra_need = sum(extra_needs)

    if total_extra_need == 0:
        return tuple(mins)

    assigned = []
    for extra_need, min_w in zip(extra_needs, mins):
        portion = int(extra_available * (extra_need / total_extra_need))
        assigned.append(min_w + portion)

    diff = available - sum(assigned)
    i = 0
    while diff > 0:
        assigned[i % 3] += 1
        diff -= 1
        i += 1

    return tuple(assigned)

def health_score(r: Result, prev_queue: int) -> int:
    score = 100
    q_th = CLUSTER_THRESHOLDS[r.cluster]

    if r.encolados >= q_th["critical"]:
        score -= 35
    elif r.encolados >= q_th["warning"]:
        score -= 20

    if r.cpu >= ADV_THRESHOLDS["cpu_critical"]:
        score -= 20
    elif r.cpu >= ADV_THRESHOLDS["cpu_warning"]:
        score -= 10

    if r.ram >= ADV_THRESHOLDS["ram_critical"]:
        score -= 15
    elif r.ram >= ADV_THRESHOLDS["ram_warning"]:
        score -= 8

    if r.conn_out >= ADV_THRESHOLDS["conn_critical"]:
        score -= 10
    elif r.conn_out >= ADV_THRESHOLDS["conn_warning"]:
        score -= 5

    growth = r.encolados - prev_queue
    if growth >= ADV_THRESHOLDS["growth_critical"]:
        score -= 15
    elif growth >= ADV_THRESHOLDS["growth_warning"]:
        score -= 8

    if r.active_recips >= ADV_THRESHOLDS["active_recips_warning"]:
        score -= 7

    if r.error or str(r.status).lower() != "online":
        score = min(score, 20)

    return max(0, min(100, score))

def diagnose_node(r: Result, prev_queue: int) -> str:
    growth = r.encolados - prev_queue

    if r.error:
        return "Sin respuesta"
    if str(r.status).lower() != "online":
        return f"Status={r.status}"
    if r.encolados >= ADV_THRESHOLDS["stuck_min_queue"] and r.conn_out <= ADV_THRESHOLDS["smtp_low_conn_out"]:
        return "Queue sin salida"
    if growth >= ADV_THRESHOLDS["growth_critical"] and r.conn_out >= ADV_THRESHOLDS["smtp_high_conn_out"]:
        return "Queue creciendo"
    if r.cpu >= ADV_THRESHOLDS["cpu_critical"]:
        return "CPU alta"
    if r.ram >= ADV_THRESHOLDS["ram_critical"]:
        return "RAM alta"
    if r.conn_out >= ADV_THRESHOLDS["conn_critical"]:
        return "Salida alta"
    if r.active_recips >= ADV_THRESHOLDS["active_recips_warning"] and r.encolados >= 50:
        return "Muchos recips"
    if r.encolados == 0 and r.conn_in == 0 and r.conn_out == 0:
        return "Sin actividad"
    return "OK"

def natural_risk_text(cluster: str, nodo: str, diag: str) -> str:
    if diag == "Sin respuesta":
        return f"{RED}{nodo} de {cluster} sin respuesta{RESET}"

    mapping = {
        "Salida alta": f"{YELLOW}Salida SMTP alta en {nodo} ({cluster}){RESET}",
        "CPU alta": f"{YELLOW}CPU alta en {nodo} ({cluster}){RESET}",
        "RAM alta": f"{YELLOW}RAM alta en {nodo} ({cluster}){RESET}",
        "Queue sin salida": f"{YELLOW}Cola sin salida en {nodo} ({cluster}){RESET}",
        "Queue creciendo": f"{YELLOW}Cola creciendo en {nodo} ({cluster}){RESET}",
        "Muchos recips": f"{YELLOW}Muchos destinatarios en {nodo} ({cluster}){RESET}",
        "Sin actividad": f"{YELLOW}Sin actividad en {nodo} ({cluster}){RESET}",
    }

    if diag.startswith("Status="):
        status = diag.split("=", 1)[1]
        return f"{YELLOW}{nodo} de {cluster}: status {status}{RESET}"

    return mapping.get(diag, f"{YELLOW}{diag} en {nodo} ({cluster}){RESET}")


def safe_cluster_filename(cluster: str) -> str:
    return (
        cluster.lower()
        .replace("á", "a")
        .replace("é", "e")
        .replace("í", "i")
        .replace("ó", "o")
        .replace("ú", "u")
        .replace(" ", "_")
    )


def cluster_rate_tendency(
    msg_in_1m: int,
    msg_in_5m: int,
    comp_1m: int,
    comp_5m: int,
    comp_15m: int,
    queue_delta: int,
    active_recips: int,
    bounce_1m: int,
) -> str:
    if queue_delta >= ADV_THRESHOLDS["growth_critical"]:
        return f"{RED}Cola creciendo{RESET}"
    if queue_delta <= -ADV_THRESHOLDS["growth_warning"]:
        return f"{GREEN}Cola drenando{RESET}"

    if active_recips >= ADV_THRESHOLDS["active_recips_warning"] * 2:
        if comp_1m < comp_5m:
            return f"{RED}Carga alta con proceso bajo{RESET}"
        return f"{YELLOW}Carga activa elevada{RESET}"

    if bounce_1m >= 1000:
        return f"{YELLOW}Rebotes temporales altos{RESET}"

    up_5 = comp_1m / max(comp_5m, 1)
    up_15 = comp_1m / max(comp_15m, 1)

    if up_5 >= 1.20 and up_15 >= 1.20:
        return f"{GREEN}Procesamiento acelerando{RESET}"

    if up_5 <= 0.80 and up_15 <= 0.80:
        return f"{RED}Procesamiento cayendo{RESET}"

    msg_ratio = msg_in_1m / max(msg_in_5m, 1)

    if msg_ratio >= 1.20 and queue_delta >= 0:
        return f"{YELLOW}Tráfico creciendo{RESET}"

    if msg_ratio <= 0.80 and queue_delta <= 0:
        return f"{GREEN}Entrada moderándose{RESET}"

    if 0.95 <= up_5 <= 1.05 and 0.95 <= up_15 <= 1.05 and abs(queue_delta) <= 5:
        return "Carga estable"

    return "Actividad irregular"

def ensure_csv_dir() -> None:
    os.makedirs(CSV_OUTPUT_DIR, exist_ok=True)


def build_cluster_csv_path(cluster: str, now: datetime) -> str:
    filename = f"{now.strftime('%Y_%m')}_{safe_cluster_filename(cluster)}.csv"
    return os.path.join(CSV_OUTPUT_DIR, filename)


def append_cluster_csv(cluster: str, results_map: Dict[str, Result], now: datetime) -> None:
    ensure_csv_dir()
    path = build_cluster_csv_path(cluster, now)
    file_exists = os.path.exists(path)

    with open(path, "a", newline="", encoding="utf-8") as f:
        writer = csv.writer(f, delimiter=";")
        if not file_exists:
            writer.writerow(CSV_HEADERS)

        for nodo in sorted(results_map.keys()):
            r = results_map[nodo]
            writer.writerow([
                now.strftime("%Y-%m-%d"),
                now.strftime("%H:%M:%S"),
                r.timestamp,
                r.cluster,
                r.nodo,
                r.status,
                r.encolados,
                r.cpu,
                r.ram,
                r.conn_in,
                r.conn_out,
                r.active_recips,
                r.inj_msgs_1m,
                r.inj_msgs_5m,
                r.inj_msgs_15m,
                r.delivered_recips_1m,
                r.delivered_recips_5m,
                r.delivered_recips_15m,
                r.inj_recips_1m,
                r.inj_recips_5m,
                r.inj_recips_15m,
                r.completed_recips_1m,
                r.completed_recips_5m,
                r.completed_recips_15m,
                r.soft_bounced_evts_1m,
                r.soft_bounced_evts_5m,
                r.soft_bounced_evts_15m,
                r.error or "",
            ])


def persist_current_data_to_csv(current_data: Dict[str, Dict[str, Result]], now: datetime) -> None:
    for cluster, results_map in current_data.items():
        append_cluster_csv(cluster, results_map, now)

def esc_listener() -> None:
    while not stop_event.is_set():
        if msvcrt.kbhit():
            key = msvcrt.getch()
            if key == b"\x1b":
                stop_event.set()
                return
        time.sleep(0.1)

def parse_status(xml_text: str) -> Tuple[str, Dict[str, int]]:
    root = ET.fromstring(xml_text)

    system_status = "unknown"
    sys_node = root.find(".//system")
    if sys_node is not None:
        system_status = sys_node.attrib.get("status", system_status)

    gauges = {
        "msgs_in_work_queue": 0,
        "cpu_utilization": 0,
        "ram_utilization": 0,
        "conn_in": 0,
        "conn_out": 0,
        "active_recips": 0,
    }

    for gauge in root.findall(".//gauge"):
        name = gauge.attrib.get("name")
        if name in gauges:
            gauges[name] = to_int(gauge.attrib.get("current"))

    rate_names = [
        "inj_msgs",
        "inj_recips",
        "soft_bounced_evts",
        "completed_recips",
        "delivered_recips",
    ]

    rates: Dict[str, int] = {}
    for name in rate_names:
        rates[f"{name}_1m"] = 0
        rates[f"{name}_5m"] = 0
        rates[f"{name}_15m"] = 0

    for rate in root.findall(".//rate"):
        name = rate.attrib.get("name")
        if name in rate_names:
            rates[f"{name}_1m"] = to_int(rate.attrib.get("last_1_min"))
            rates[f"{name}_5m"] = to_int(rate.attrib.get("last_5_min"))
            rates[f"{name}_15m"] = to_int(rate.attrib.get("last_15_min"))

    metrics = {**gauges, **rates}
    return system_status, metrics

def fetch_one(
    session: requests.Session,
    url: str,
    auth: Tuple[str, str],
    timestamp: str,
    cluster: str,
    idx: int,
) -> Result:
    nodo = nodo_from_index(idx)
    last_err: Optional[str] = None

    for _ in range(RETRIES + 1):
        try:
            r = session.get(
                url,
                auth=auth,
                timeout=TIMEOUT_SECONDS,
                verify=VERIFY_SSL,
                headers={"Accept": "application/xml"},
            )

            if r.status_code != 200:
                return Result(
                    timestamp=timestamp,
                    nodo=nodo,
                    cluster=cluster,
                    status="offline",
                    encolados=0,
                    cpu=0,
                    ram=0,
                    conn_in=0,
                    conn_out=0,
                    active_recips=0,
                    inj_msgs_1m=0,
                    inj_msgs_5m=0,
                    inj_msgs_15m=0,
                    delivered_recips_1m=0,
                    delivered_recips_5m=0,
                    delivered_recips_15m=0,
                    inj_recips_1m=0,
                    inj_recips_5m=0,
                    inj_recips_15m=0,
                    completed_recips_1m=0,
                    completed_recips_5m=0,
                    completed_recips_15m=0,
                    soft_bounced_evts_1m=0,
                    soft_bounced_evts_5m=0,
                    soft_bounced_evts_15m=0,
                    error=f"HTTP {r.status_code}",
                )

            status, m = parse_status(r.text)

            return Result(
                timestamp=timestamp,
                nodo=nodo,
                cluster=cluster,
                status=status,
                encolados=m["msgs_in_work_queue"],
                cpu=m["cpu_utilization"],
                ram=m["ram_utilization"],
                conn_in=m["conn_in"],
                conn_out=m["conn_out"],
                active_recips=m["active_recips"],
                inj_msgs_1m=m["inj_msgs_1m"],
                inj_msgs_5m=m["inj_msgs_5m"],
                inj_msgs_15m=m["inj_msgs_15m"],
                delivered_recips_1m=m["delivered_recips_1m"],
                delivered_recips_5m=m["delivered_recips_5m"],
                delivered_recips_15m=m["delivered_recips_15m"],
                inj_recips_1m=m["inj_recips_1m"],
                inj_recips_5m=m["inj_recips_5m"],
                inj_recips_15m=m["inj_recips_15m"],
                completed_recips_1m=m["completed_recips_1m"],
                completed_recips_5m=m["completed_recips_5m"],
                completed_recips_15m=m["completed_recips_15m"],
                soft_bounced_evts_1m=m["soft_bounced_evts_1m"],
                soft_bounced_evts_5m=m["soft_bounced_evts_5m"],
                soft_bounced_evts_15m=m["soft_bounced_evts_15m"],
                error=None,
            )

        except requests.exceptions.RequestException as e:
            last_err = str(e)

    return Result(
        timestamp=timestamp,
        nodo=nodo,
        cluster=cluster,
        status="offline",
        encolados=0,
        cpu=0,
        ram=0,
        conn_in=0,
        conn_out=0,
        active_recips=0,
        inj_msgs_1m=0,
        inj_msgs_5m=0,
        inj_msgs_15m=0,
        delivered_recips_1m=0,
        delivered_recips_5m=0,
        delivered_recips_15m=0,
        inj_recips_1m=0,
        inj_recips_5m=0,
        inj_recips_15m=0,
        completed_recips_1m=0,
        completed_recips_5m=0,
        completed_recips_15m=0,
        soft_bounced_evts_1m=0,
        soft_bounced_evts_5m=0,
        soft_bounced_evts_15m=0,
        error=last_err or "RequestException",
    )


def fetch_cluster(
    session: requests.Session,
    group: Dict,
    auth: Tuple[str, str],
    timestamp: str,
) -> List[Result]:
    cluster = group["cluster"]
    pattern = group["pattern"]

    results: List[Result] = []
    with ThreadPoolExecutor(max_workers=MAX_WORKERS) as ex:
        futures = [
            ex.submit(fetch_one, session, pattern.format(idx=i), auth, timestamp, cluster, i)
            for i in range(1, 21)
        ]
        for fut in as_completed(futures):
            results.append(fut.result())

    return sorted(results, key=lambda r: r.nodo)

def build_status_incidents(current_data: Dict[str, Dict[str, Result]]) -> List[Tuple[str, str, str]]:
    incidents: List[Tuple[str, str, str]] = []

    for cluster, nodo_map in current_data.items():
        for nodo, r in nodo_map.items():
            if r.error:
                incidents.append((cluster, nodo, "Sin respuesta"))
            elif str(r.status).lower() != "online":
                incidents.append((cluster, nodo, f"Status={r.status}"))

    incidents.sort(key=lambda x: (x[0], x[1]))
    return incidents


def detect_mail_backlog_real(current_data: Dict[str, Dict[str, Result]]) -> List[str]:
    findings = []

    for cluster, nodo_map in current_data.items():
        total_queue = sum(r.encolados for r in nodo_map.values())
        total_out = sum(r.conn_out for r in nodo_map.values())

        th = CLUSTER_THRESHOLDS[cluster]
        if total_queue >= th["total_warning"] and total_out <= len(nodo_map) * ADV_THRESHOLDS["smtp_low_conn_out"]:
            findings.append(
                f"{cluster}: backlog real (cola={fmt_num_dot(total_queue)}, conn_out={fmt_num_dot(total_out)})"
            )

    return findings


def detect_smtp_saturation(current_data: Dict[str, Dict[str, Result]]) -> List[str]:
    findings = []

    for cluster, nodo_map in current_data.items():
        for nodo, r in nodo_map.items():
            if r.conn_out >= ADV_THRESHOLDS["conn_critical"] and r.cpu >= ADV_THRESHOLDS["cpu_warning"]:
                findings.append(f"{cluster} {nodo}: saturación SMTP (CO={fmt_num_dot(r.conn_out)}, CPU={r.cpu})")
            elif r.conn_out >= ADV_THRESHOLDS["conn_critical"] and r.ram >= ADV_THRESHOLDS["ram_warning"]:
                findings.append(f"{cluster} {nodo}: salida alta con RAM elevada (CO={fmt_num_dot(r.conn_out)}, RAM={r.ram})")

    return findings


def detect_cluster_imbalance(current_data: Dict[str, Dict[str, Result]]) -> List[str]:
    findings = []

    for cluster, nodo_map in current_data.items():
        queues = [r.encolados for r in nodo_map.values()]
        if not queues:
            continue

        max_q = max(queues)
        min_q = min(queues)
        avg_q = sum(queues) / len(queues)

        if avg_q <= 0:
            continue

        ratio = max_q / max(avg_q, 1)

        if ratio >= ADV_THRESHOLDS["imbalance_ratio_critical"]:
            findings.append(f"{cluster}: desbalanceado crítico (máx={fmt_num_dot(max_q)}, media={fmt_num_dot(int(avg_q))})")
        elif ratio >= ADV_THRESHOLDS["imbalance_ratio_warning"]:
            findings.append(f"{cluster}: desbalanceado (máx={fmt_num_dot(max_q)}, media={fmt_num_dot(int(avg_q))})")

        if max_q > 0 and min_q == 0:
            findings.append(f"{cluster}: nodos vacíos y nodos cargados a la vez")

    return findings


def detect_queue_storm(
    current_data: Dict[str, Dict[str, Result]],
    previous_data: Dict[str, Dict[str, int]],
) -> List[str]:
    findings = []

    for cluster, nodo_map in current_data.items():
        for nodo, r in nodo_map.items():
            prev_q = previous_data.get(cluster, {}).get(nodo, r.encolados)
            growth = r.encolados - prev_q

            if growth >= ADV_THRESHOLDS["growth_critical"]:
                findings.append(f"{cluster} {nodo}: queue storm +{fmt_num_dot(growth)}")
            elif growth >= ADV_THRESHOLDS["growth_warning"] and r.conn_out <= ADV_THRESHOLDS["smtp_low_conn_out"]:
                findings.append(f"{cluster} {nodo}: cola subiendo sin salida +{fmt_num_dot(growth)}")

    return findings


def detect_stuck_nodes(history: Dict[str, Dict[str, List[int]]]) -> List[str]:
    findings = []

    for cluster, nodo_map in history.items():
        for nodo, series in nodo_map.items():
            if len(series) < ADV_THRESHOLDS["stuck_cycles"]:
                continue

            tail = series[-ADV_THRESHOLDS["stuck_cycles"]:]
            if len(set(tail)) == 1 and tail[-1] >= ADV_THRESHOLDS["stuck_min_queue"]:
                findings.append(f"{cluster} {nodo}: nodo stuck (cola fija={fmt_num_dot(tail[-1])})")

    return findings


def build_advanced_findings(
    current_data: Dict[str, Dict[str, Result]],
    previous_data: Dict[str, Dict[str, int]],
    history: Dict[str, Dict[str, List[int]]],
) -> List[str]:
    lines: List[str] = []

    backlog = detect_mail_backlog_real(current_data)
    saturation = detect_smtp_saturation(current_data)
    imbalance = detect_cluster_imbalance(current_data)
    storms = detect_queue_storm(current_data, previous_data)
    stuck = detect_stuck_nodes(history)

    if not any([backlog, saturation, imbalance, storms, stuck]):
        lines.append(f"{GREEN}Sin hallazgos relevantes.{RESET}")
        lines.append("La plataforma no muestra señales claras de congestión o desequilibrio.")
        return lines

    if imbalance:
        lines.append(f"{YELLOW}Desbalanceo entre nodos del cluster:{RESET}")
        for item in imbalance[:5]:
            lines.append(f"  - {item}")
        lines.append("")

    if storms:
        lines.append(f"{RED}Crecimiento brusco de cola:{RESET}")
        for item in storms[:5]:
            lines.append(f"  - {item}")
        lines.append("")

    if backlog:
        lines.append(f"{RED}Backlog real detectado:{RESET}")
        for item in backlog[:5]:
            lines.append(f"  - {item}")
        lines.append("")

    if saturation:
        lines.append(f"{YELLOW}Saturación SMTP:{RESET}")
        for item in saturation[:5]:
            lines.append(f"  - {item}")
        lines.append("")

    if stuck:
        lines.append(f"{RED}Nodos posiblemente bloqueados:{RESET}")
        for item in stuck[:5]:
            lines.append(f"  - {item}")

    return lines

def build_cluster_summary_lines(
    current_data: Dict[str, Dict[str, Result]],
    previous_data: Dict[str, Dict[str, int]],
) -> List[str]:
    nodos = [f"ESA{i:02d}" for i in range(1, 21)]

    c1 = 13
    c2 = 8
    c3 = 10
    c4 = 18

    lines: List[str] = []
    lines.append("RESUMEN CLUSTERS")
    lines.append("─" * (c1 + c2 + c3 + c4 + 9))
    lines.append(
        f"{'Cluster':<{c1}} | "
        f"{'Total':^{c2}} | "
        f"{'Estado':^{c3}} | "
        f"{'Última variación':^{c4}}"
    )

    for group in GROUPS:
        cluster = group["cluster"]
        current_total = sum(r.encolados for r in current_data.get(cluster, {}).values())
        prev_map = previous_data.get(cluster, {})
        previous_total = sum(prev_map.get(nodo, 0) for nodo in nodos)

        if cluster not in previous_data:
            variation_text = "Sin histórico"
        else:
            delta = current_total - previous_total
            if delta > 0:
                variation_text = f"Sube +{fmt_num_dot(delta)}"
            elif delta < 0:
                variation_text = f"Baja -{fmt_num_dot(abs(delta))}"
            else:
                variation_text = "Sin cambios"

        status_text = total_status_colored(cluster, current_total)

        lines.append(
            f"{cluster:<{c1}} | "
            f"{fmt_num_dot(current_total):>{c2}} | "
            f"{ansi_center(status_text, c3)} | "
            f"{variation_text:^{c4}}"
        )

    return lines


def build_alert_column_lines(
    current_data: Dict[str, Dict[str, Result]],
) -> List[str]:
    lines: List[str] = []
    lines.append("ALERTAS NODOS")
    lines.append("─" * 42)

    found = False
    for cluster, nodo_map in current_data.items():
        th = CLUSTER_THRESHOLDS[cluster]

        for nodo, r in nodo_map.items():
            if r.error:
                lines.append(natural_alert_text(cluster, nodo, r, "error"))
                found = True
            elif r.encolados >= th["critical"]:
                lines.append(natural_alert_text(cluster, nodo, r, "critical"))
                found = True
            elif r.encolados >= th["warning"]:
                lines.append(natural_alert_text(cluster, nodo, r, "warning"))
                found = True

    if not found:
        lines.append(f"{GREEN}No se han detectado alertas{RESET}")

    return lines


def build_risk_column_lines(
    current_data: Dict[str, Dict[str, Result]],
    previous_data: Dict[str, Dict[str, int]],
) -> List[str]:
    lines: List[str] = []
    lines.append("NODOS EN RIESGO")
    lines.append("─" * 42)

    found = False
    for cluster, nodo_map in current_data.items():
        for nodo, r in nodo_map.items():
            prev_q = previous_data.get(cluster, {}).get(nodo, r.encolados)
            diag = diagnose_node(r, prev_q)

            if diag != "OK":
                lines.append(natural_risk_text(cluster, nodo, diag))
                found = True

    if not found:
        lines.append(f"{GREEN}No se han detectado nodos en riesgo{RESET}")

    return lines


def print_top_three_columns(
    current_data: Dict[str, Dict[str, Result]],
    previous_data: Dict[str, Dict[str, int]],
) -> None:
    gap = "   "

    col1 = build_cluster_summary_lines(current_data, previous_data)
    col2 = build_alert_column_lines(current_data)
    col3 = build_risk_column_lines(current_data, previous_data)

    col1_w, col2_w, col3_w = choose_three_col_widths(
        col1, col2, col3,
        min1=58,
        min2=40,
        min3=40,
        gap=len(gap),
    )

    rows = max(len(col1), len(col2), len(col3))

    col1 += [""] * (rows - len(col1))
    col2 += [""] * (rows - len(col2))
    col3 += [""] * (rows - len(col3))

    for a, b, c in zip(col1, col2, col3):
        print(
            f"{pad_or_trim(a, col1_w)}{gap}"
            f"{pad_or_trim(b, col2_w)}{gap}"
            f"{pad_or_trim(c, col3_w)}"
        )

def print_rates_table(
    current_data: Dict[str, Dict[str, Result]],
    previous_data: Dict[str, Dict[str, int]],
) -> None:
    print("RENDIMIENTO OPERATIVO POR CLUSTER\n")

    term_w = terminal_width(180)
    nodos = [f"ESA{i:02d}" for i in range(1, 21)]

    headers = [
        ("cluster", "Cluster"),
        ("in", "MSG In"),
        ("out", "Recipient Out"),
        ("rec", "Recipient In"),
        ("comp", "Recipient Proc"),
        ("active", "Active Rcpt"),
        ("bnc", "Bounce Temp"),
        ("cola", "Cola"),
        ("trend", "Tendencia"),
    ]

    rows = []
    for group in GROUPS:
        cluster = group["cluster"]
        nodo_map = current_data.get(cluster, {})
        prev_map = previous_data.get(cluster, {})

        total_in_1m = sum(r.inj_msgs_1m for r in nodo_map.values())
        total_in_5m = sum(r.inj_msgs_5m for r in nodo_map.values())

        total_out_1m = sum(r.delivered_recips_1m for r in nodo_map.values())
        total_rec_1m = sum(r.inj_recips_1m for r in nodo_map.values())

        total_comp_1m = sum(r.completed_recips_1m for r in nodo_map.values())
        total_comp_5m = sum(r.completed_recips_5m for r in nodo_map.values())
        total_comp_15m = sum(r.completed_recips_15m for r in nodo_map.values())

        total_active = sum(r.active_recips for r in nodo_map.values())
        total_bnc_1m = sum(r.soft_bounced_evts_1m for r in nodo_map.values())

        total_cur_q = sum(r.encolados for r in nodo_map.values())
        total_prev_q = sum(
            prev_map.get(n, nodo_map.get(n).encolados if nodo_map.get(n) else 0)
            for n in nodos
        )
        delta_q = total_cur_q - total_prev_q

        if delta_q >= ADV_THRESHOLDS["growth_critical"]:
            cola_txt = f"{RED}+{fmt_num_dot(delta_q)} creciendo{RESET}"
        elif delta_q >= ADV_THRESHOLDS["growth_warning"]:
            cola_txt = f"{YELLOW}+{fmt_num_dot(delta_q)} subiendo{RESET}"
        elif delta_q <= -ADV_THRESHOLDS["growth_critical"]:
            cola_txt = f"{GREEN}{fmt_num_dot(delta_q)} drenando{RESET}"
        elif delta_q <= -ADV_THRESHOLDS["growth_warning"]:
            cola_txt = f"{GREEN}{fmt_num_dot(delta_q)} bajando{RESET}"
        else:
            cola_txt = "Estable"

        active_txt = color_by_threshold(
            fmt_num_dot(total_active),
            total_active,
            ADV_THRESHOLDS["active_recips_warning"],
            ADV_THRESHOLDS["active_recips_warning"] * 2,
        )

        tendencia_txt = cluster_rate_tendency(
            msg_in_1m=total_in_1m,
            msg_in_5m=total_in_5m,
            comp_1m=total_comp_1m,
            comp_5m=total_comp_5m,
            comp_15m=total_comp_15m,
            queue_delta=delta_q,
            active_recips=total_active,
            bounce_1m=total_bnc_1m,
        )

        rows.append({
            "cluster": cluster,
            "in": fmt_num_dot(total_in_1m),
            "out": fmt_num_dot(total_out_1m),
            "rec": fmt_num_dot(total_rec_1m),
            "comp": fmt_num_dot(total_comp_1m),
            "active": active_txt,
            "bnc": fmt_num_dot(total_bnc_1m),
            "cola": cola_txt,
            "trend": tendencia_txt,
        })

    natural_widths = {}
    for key, title in headers:
        max_len = len(title)
        for row in rows:
            max_len = max(max_len, visible_text_len(row[key]))
        natural_widths[key] = max_len + 2

    total_natural = sum(natural_widths[k] for k, _ in headers)

    if total_natural <= term_w:
        widths = natural_widths.copy()
    else:
        min_widths = {
            "cluster": 10,
            "in": 8,
            "out": 10,
            "rec": 10,
            "comp": 10,
            "active": 10,
            "bnc": 9,
            "cola": 12,
            "trend": 18,
        }

        base_total = sum(min_widths.values())
        available_extra = max(0, term_w - base_total)

        extras_needed = {
            k: max(0, natural_widths[k] - min_widths[k])
            for k, _ in headers
        }
        extras_total = sum(extras_needed.values())

        widths = min_widths.copy()

        if extras_total > 0 and available_extra > 0:
            allocated = {}
            for k, _ in headers:
                share = int(available_extra * (extras_needed[k] / extras_total))
                allocated[k] = share
                widths[k] += share

            used = sum(allocated.values())
            remainder = available_extra - used

            priority = sorted(
                [k for k, _ in headers],
                key=lambda x: extras_needed[x],
                reverse=True
            )
            i = 0
            while remainder > 0 and priority:
                widths[priority[i % len(priority)]] += 1
                remainder -= 1
                i += 1

    header = "".join(
        title.ljust(widths[key]) if key == "cluster"
        else title.rjust(widths[key])
        for key, title in headers
    )
    print(header)
    print("-" * visible_text_len(header))

    for row in rows:
        line = (
            f"{row['cluster']:<{widths['cluster']}}"
            f"{ansi_rjust(row['in'], widths['in'])}"
            f"{ansi_rjust(row['out'], widths['out'])}"
            f"{ansi_rjust(row['rec'], widths['rec'])}"
            f"{ansi_rjust(row['comp'], widths['comp'])}"
            f"{ansi_rjust(row['active'], widths['active'])}"
            f"{ansi_rjust(row['bnc'], widths['bnc'])}"
            f"{ansi_rjust(row['cola'], widths['cola'])}"
            f"{ansi_rjust(row['trend'], widths['trend'])}"
        )
        print(line)


def print_status_incidents(current_data: Dict[str, Dict[str, Result]]) -> None:
    incidents = build_status_incidents(current_data)

    print("INCIDENCIAS\n")

    if not incidents:
        print(f"{GREEN}No se han detectado incidencias{RESET}")
        return

    for cluster, nodo, msg in incidents:
        if msg == "Sin respuesta":
            print(f"{RED}No hay respuesta del nodo {nodo} de {cluster}{RESET}")
        else:
            status = msg.split("=", 1)[1] if "=" in msg else msg
            print(f"{RED}El estado del nodo {nodo} de {cluster} es {status}{RESET}")

def run_monitor(auth: Tuple[str, str]) -> None:
    previous_data: Dict[str, Dict[str, int]] = {}
    queue_history: Dict[str, Dict[str, List[int]]] = {}

    listener = threading.Thread(target=esc_listener, daemon=True)
    listener.start()

    with requests.Session() as session:
        while not stop_event.is_set():
            now = datetime.now()
            timestamp = now.strftime("%H:%M")
            current_data: Dict[str, Dict[str, Result]] = {}

            with ThreadPoolExecutor(max_workers=len(GROUPS)) as ex:
                fut_map = {
                    ex.submit(fetch_cluster, session, g, auth, timestamp): g["cluster"]
                    for g in GROUPS
                }

                for fut in as_completed(fut_map):
                    cluster = fut_map[fut]
                    results = fut.result()
                    current_data[cluster] = {r.nodo: r for r in results}

            persist_current_data_to_csv(current_data, now)

            for cluster, nodo_map in current_data.items():
                queue_history.setdefault(cluster, {})
                for nodo, r in nodo_map.items():
                    queue_history[cluster].setdefault(nodo, []).append(r.encolados)
                    queue_history[cluster][nodo] = queue_history[cluster][nodo][-5:]

            previous_data_snapshot = {
                c: {n: r.encolados for n, r in m.items()}
                for c, m in current_data.items()
            }

            next_refresh_time = datetime.now() + timedelta(seconds=REFRESH_SECONDS)

            def draw():
                print_header_with_timer("MONITOR COLAS ESA", next_refresh_time)

                print_top_three_columns(current_data, previous_data)
                print()

                print_orange_separator()
                print()
                print_queue_table(timestamp, current_data, previous_data)

                print()
                print_orange_separator()
                print()
                print_node_health_table(current_data, previous_data)

                print()
                print_orange_separator()
                print()
                print_rates_table(current_data, previous_data)

                print()
                print_orange_separator()
                print()
                print_status_incidents(current_data)

                print()
                print_orange_separator()
                print()
                print_advanced_findings(current_data, previous_data, queue_history)

                print(f"\nCSV mensuales en: {CSV_OUTPUT_DIR}")
                print("Pulsa ESC para salir")

            remaining = REFRESH_SECONDS
            while remaining > 0 and not stop_event.is_set():
                render_screen(draw)

                sleep_time = min(COUNTDOWN_STEP_SECONDS, remaining)
                slept = 0.0
                while slept < sleep_time and not stop_event.is_set():
                    time.sleep(0.1)
                    slept += 0.1

                remaining -= sleep_time

            previous_data = previous_data_snapshot

    clear_screen()
    print("Monitor detenido (ESC pulsado)")


def main() -> None:
    creds = load_env_file(CREDENTIALS_FILE)
    auth = (creds["GLOBAL_USERNAME"], creds["GLOBAL_PASSWORD"])

    ensure_csv_dir()
    clear_screen()
    run_monitor(auth)


if __name__ == "__main__":
    main()
