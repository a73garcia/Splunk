def clear_screen() -> None:

def load_env_file(path: str) -> Dict[str, str]:
    data: Dict[str, str] = {}
    with open(path, "r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line or line.startswith("#") or "=" not in line:
                continue
            k, v = line.split("=", 1)
            data[k.strip()] = v.strip()
    return data


def nodo_from_index(idx: int) -> str:
    return f"ESA{idx:02d}"


def fmt_num_dot(n: int) -> str:
    return f"{n:,}".replace(",", ".")


def to_int(value: Optional[str]) -> int:
    if value is None:
        return 0
    s = str(value).strip()
    if not s or s.upper() == "N/A":
        return 0

    digits = []
    for ch in s:
        if ch.isdigit() or ch == ".":
            digits.append(ch)
        else:
            break

    try:
        return int(float("".join(digits))) if digits else 0
    except Exception:
        return 0


def trend_symbol(prev: int, cur: int) -> str:
    if cur > prev:
        return "↑"
    if cur < prev:
        return "↓"
    return "→"


def total_status_colored(cluster: str, total: int) -> str:
    th = CLUSTER_THRESHOLDS[cluster]

    if total >= th["total_critical"]:
        return f"{RED}{'CRITICAL':<8}{RESET}"
    if total >= th["total_warning"]:
        return f"{YELLOW}{'WARNING ':<8}{RESET}"
    return f"{GREEN}{'OK':^8}{RESET}"


def print_orange_separator(width: int = 220) -> None:
    line = "═" * width
    print(f"{ORANGE}{line}{RESET}")


def orange_cluster_sep() -> str:
    return f"{ORANGE}||{RESET}"


def color_by_threshold(text: str, value: int, warning: int, critical: int) -> str:
    if value >= critical:
        return f"{RED}{text}{RESET}"
    if value >= warning:
        return f"{YELLOW}{text}{RESET}"
    return text


def queue_color(cluster: str, text: str, value: int) -> str:
    th = CLUSTER_THRESHOLDS[cluster]
    return color_by_threshold(text, value, th["warning"], th["critical"])


def health_score(r: Result, prev_queue: int) -> int:
    score = 100

    q_th = CLUSTER_THRESHOLDS[r.cluster]
    if r.encolados >= q_th["critical"]:
        score -= 35
    elif r.encolados >= q_th["warning"]:
        score -= 20

    if r.cpu >= ADV_THRESHOLDS["cpu_critical"]:
        score -= 20
    elif r.cpu >= ADV_THRESHOLDS["cpu_warning"]:
        score -= 10

    if r.ram >= ADV_THRESHOLDS["ram_critical"]:
        score -= 15
    elif r.ram >= ADV_THRESHOLDS["ram_warning"]:
        score -= 8

    if r.conn_out >= ADV_THRESHOLDS["conn_critical"]:
        score -= 10
    elif r.conn_out >= ADV_THRESHOLDS["conn_warning"]:
        score -= 5

    growth = r.encolados - prev_queue
    if growth >= ADV_THRESHOLDS["growth_critical"]:
        score -= 15
    elif growth >= ADV_THRESHOLDS["growth_warning"]:
        score -= 8

    if r.active_recips >= ADV_THRESHOLDS["active_recips_warning"]:
        score -= 7

    if r.error:
        score = min(score, 20)

    return max(0, min(100, score))


def health_color(text: str, score: int) -> str:
    if score < 60:
        return f"{RED}{text}{RESET}"
    if score < 80:
        return f"{YELLOW}{text}{RESET}"
    return f"{GREEN}{text}{RESET}"


def diagnose_node(r: Result, prev_queue: int) -> str:
    growth = r.encolados - prev_queue

    if r.error:
        return "Sin respuesta"
    if r.status.lower() != "online":
        return f"Status {r.status}"
    if r.encolados >= 100 and r.conn_out <= 5:
        return "Queue sin salida"
    if growth >= 75 and r.conn_out >= 20:
        return "Queue creciendo"
    if r.cpu >= 85:
        return "CPU alta"
    if r.ram >= 90:
        return "RAM alta"
    if r.conn_out >= 40:
        return "Salida alta"
    if r.active_recips >= 5000 and r.encolados >= 50:
        return "Muchos recips"
    if r.encolados == 0 and r.conn_in == 0 and r.conn_out == 0:
        return "Sin actividad"
    return "OK"


def status_incident_symbol(status: str, error: Optional[str]) -> str:
    if error:
        return f"{RED}!{RESET}"
    if str(status).lower() != "online":
        return f"{YELLOW}!{RESET}"
    return " "


def build_status_incidents(current_data: Dict[str, Dict[str, Result]]) -> List[Tuple[str, str, str]]:
    incidents: List[Tuple[str, str, str]] = []
    for cluster, nodo_map in current_data.items():
        for nodo, r in nodo_map.items():
            if r.error:
                incidents.append((cluster, nodo, "Sin respuesta"))
            elif str(r.status).lower() != "online":
                incidents.append((cluster, nodo, f"Status={r.status}"))
    incidents.sort(key=lambda x: (x[0], x[1]))
    return incidents

def esc_listener() -> None:
    while not stop_event.is_set():
        if msvcrt.kbhit():
            key = msvcrt.getch()
            if key == b"\x1b":
                stop_event.set()
                return
        time.sleep(0.1)

def parse_status(xml_text: str) -> Tuple[str, Dict[str, int]]:
    root = ET.fromstring(xml_text)

    system_status = "No disponible"
    sys_node = root.find(".//system")
    if sys_node is not None:
        system_status = sys_node.attrib.get("status", system_status)

    metrics = {
        "msgs_in_work_queue": 0,
        "cpu_utilization": 0,
        "ram_utilization": 0,
        "conn_in": 0,
        "conn_out": 0,
        "active_recips": 0,
    }

    for gauge in root.findall(".//gauge"):
        name = gauge.attrib.get("name")
        if name in metrics:
            metrics[name] = to_int(gauge.attrib.get("current"))

    return system_status, metrics

def fetch_one(
    session: requests.Session,
    url: str,
    auth: Tuple[str, str],
    timestamp: str,
    cluster: str,
    idx: int,
) -> Result:
    nodo = nodo_from_index(idx)
    last_err: Optional[str] = None

    for _ in range(RETRIES + 1):
        try:
            r = session.get(
                url,
                auth=auth,
                timeout=TIMEOUT_SECONDS,
                verify=VERIFY_SSL,
                headers={"Accept": "application/xml"},
            )

            if r.status_code != 200:
                return Result(timestamp, nodo, cluster, "ERR", 0, 0, 0, 0, 0, 0, f"HTTP {r.status_code}")

            status, m = parse_status(r.text)
            return Result(
                timestamp=timestamp,
                nodo=nodo,
                cluster=cluster,
                status=status,
                encolados=m["msgs_in_work_queue"],
                cpu=m["cpu_utilization"],
                ram=m["ram_utilization"],
                conn_in=m["conn_in"],
                conn_out=m["conn_out"],
                active_recips=m["active_recips"],
                error=None,
            )

        except requests.exceptions.RequestException as e:
            last_err = str(e)

    return Result(timestamp, nodo, cluster, "ERR", 0, 0, 0, 0, 0, 0, last_err or "RequestException")


def fetch_cluster(
    session: requests.Session,
    group: Dict,
    auth: Tuple[str, str],
    timestamp: str,
) -> List[Result]:
    cluster = group["cluster"]
    pattern = group["pattern"]
    start = int(group["start"])
    end = int(group["end"])

    results: List[Result] = []
    with ThreadPoolExecutor(max_workers=MAX_WORKERS) as ex:
        futures = [
            ex.submit(fetch_one, session, pattern.format(idx=i), auth, timestamp, cluster, i)
            for i in range(start, end + 1)
        ]
        for fut in as_completed(futures):
            results.append(fut.result())

    return sorted(results, key=lambda r: r.nodo)

def build_node_alert_lines(current_data: Dict[str, Dict[str, Result]]) -> List[str]:
    alerts = []

    for cluster, nodo_map in current_data.items():
        th = CLUSTER_THRESHOLDS[cluster]
        for nodo, r in nodo_map.items():
            if r.error:
                alerts.append((f"{RED}ERROR   {RESET}", cluster, nodo, "Sin respuesta"))
            elif r.encolados >= th["critical"]:
                alerts.append((f"{RED}CRITICAL{RESET}", cluster, nodo, fmt_num_dot(r.encolados)))
            elif r.encolados >= th["warning"]:
                alerts.append((f"{YELLOW}WARNING {RESET}", cluster, nodo, fmt_num_dot(r.encolados)))

    lines = ["ALERTAS NODOS", ""]

    if not alerts:
        lines.append(f"{GREEN}OK - Sin alertas{RESET}")
        return lines

    for level, cluster, nodo, value in alerts[:18]:
        lines.append(f"{level} {cluster} {nodo} → {value}")

    return lines


def build_risk_nodes_lines(
    current_data: Dict[str, Dict[str, Result]],
    previous_data: Dict[str, Dict[str, int]],
) -> List[str]:
    lines = ["NODOS EN RIESGO", ""]
    findings = []

    for cluster_map in current_data.values():
        for r in cluster_map.values():
            prev_q = previous_data.get(r.cluster, {}).get(r.nodo, r.encolados)
            diag = diagnose_node(r, prev_q)

            if diag != "OK":
                findings.append((r.nodo, r.cluster, diag))

    if not findings:
        lines.append(f"{GREEN}OK - Sin nodos en riesgo{RESET}")
        return lines

    dedup = []
    seen = set()
    for item in findings:
        if item not in seen:
            seen.add(item)
            dedup.append(item)

    for nodo, cluster, diag in dedup[:18]:
        color = RED if diag == "Sin respuesta" else YELLOW
        lines.append(f"{color}{nodo} {cluster}: {diag}{RESET}")

    return lines


def print_two_column_panels(left_lines: List[str], right_lines: List[str], left_width: int = 58) -> None:
    total_rows = max(len(left_lines), len(right_lines))
    left = left_lines + [""] * (total_rows - len(left_lines))
    right = right_lines + [""] * (total_rows - len(right_lines))

    for l, r in zip(left, right):
        print(f"{l.ljust(left_width)}    {r}")


def print_cluster_summary_with_side_panels(
    current_data: Dict[str, Dict[str, Result]],
    previous_data: Dict[str, Dict[str, int]],
) -> None:
    nodos = [f"ESA{i:02d}" for i in range(1, 21)]

    summary_lines = ["RESUMEN CLUSTERS", ""]
    header = f"{'Cluster':<13} | {'Total':>10} | {'Estado':^8} | {'Última variación':^22}"
    summary_lines.append(header)
    summary_lines.append("-" * len(header))

    for group in GROUPS:
        cluster = group["cluster"]
        current_total = sum(r.encolados for r in current_data.get(cluster, {}).values())

        prev_map = previous_data.get(cluster, {})
        previous_total = sum(prev_map.get(nodo, 0) for nodo in nodos)

        if cluster not in previous_data:
            variation_text = "Sin histórico"
        else:
            delta = current_total - previous_total
            if delta > 0:
                variation_text = f"Sube +{fmt_num_dot(delta)}"
            elif delta < 0:
                variation_text = f"Baja -{fmt_num_dot(abs(delta))}"
            else:
                variation_text = "Sin cambios"

        status_colored = total_status_colored(cluster, current_total)

        summary_lines.append(
            f"{cluster:<13} | {fmt_num_dot(current_total):>10} | {status_colored} | {variation_text:^22}"
        )

    alert_lines = build_node_alert_lines(current_data)
    risk_lines = build_risk_nodes_lines(current_data, previous_data)

    right_lines = ["ALERTAS / RIESGO", ""]
    max_rows = max(len(alert_lines), len(risk_lines))
    a = alert_lines + [""] * (max_rows - len(alert_lines))
    r = risk_lines + [""] * (max_rows - len(risk_lines))

    for left, right in zip(a, r):
        if left and right:
            right_lines.append(f"{left}    {right}")
        elif left:
            right_lines.append(left)
        else:
            right_lines.append(right)

    print_two_column_panels(summary_lines, right_lines, left_width=66)


def print_queue_table(
    timestamp: str,
    current_data: Dict[str, Dict[str, Result]],
    previous_data: Dict[str, Dict[str, int]],
) -> None:
    print("ESTADO ENCOLADOS NODOS\n")

    cluster_order = ["Europa", "América", "Brasil", "Aplicaciones"]
    nodos = [f"ESA{i:02d}" for i in range(1, 21)]

    ts_width = 5
    nodo_w = 6
    ant_w = 9
    act_w = 9
    var_w = 5
    ac_w = 8

    col_sep = " | "
    cluster_sep_plain = " || "
    cluster_sep_colored = f" {orange_cluster_sep()} "
    cluster_chunk_width = ant_w + len(col_sep) + act_w + len(col_sep) + var_w + len(col_sep) + ac_w

    def fmt_chunk(prev_val: int, cur_val: int, var_val: str, ac_val: int, cluster: str, has_error: bool) -> str:
        ant_txt = fmt_num_dot(prev_val).rjust(ant_w)
        q_txt = queue_color(cluster, fmt_num_dot(cur_val).rjust(act_w), cur_val)
        var_txt = var_val.center(var_w)
        ac_txt = fmt_num_dot(ac_val).rjust(ac_w)

        if has_error:
            ant_txt = f"{RED}{ant_txt}{RESET}"
            q_txt = f"{RED}{fmt_num_dot(cur_val).rjust(act_w)}{RESET}"
            var_txt = f"{RED}{var_txt}{RESET}"
            ac_txt = f"{RED}{fmt_num_dot(ac_val).rjust(ac_w)}{RESET}"

        return f"{ant_txt}{col_sep}{q_txt}{col_sep}{var_txt}{col_sep}{ac_txt}"

    h1 = (
        f"| {'Hora'.ljust(ts_width)} | {'Nodo'.ljust(nodo_w)}"
        f"{cluster_sep_colored}{'Europa'.center(cluster_chunk_width)}"
        f"{cluster_sep_colored}{'América'.center(cluster_chunk_width)}"
        f"{cluster_sep_colored}{'Brasil'.center(cluster_chunk_width)}"
        f"{cluster_sep_colored}{'Aplicaciones'.center(cluster_chunk_width)} |"
    )

    sub = (
        f"{'Anterior'.center(ant_w)}{col_sep}"
        f"{'Encolados'.center(act_w)}{col_sep}"
        f"{'Var'.center(var_w)}{col_sep}"
        f"{'AC'.center(ac_w)}"
    )

    h2 = (
        f"| {'':<{ts_width}} | {'':<{nodo_w}}"
        f"{cluster_sep_colored}{sub}"
        f"{cluster_sep_colored}{sub}"
        f"{cluster_sep_colored}{sub}"
        f"{cluster_sep_colored}{sub} |"
    )

    sep = "-" * (
        2 + ts_width + 3 + nodo_w +
        (len(cluster_sep_plain) + cluster_chunk_width) * 4 + 2
    )

    print(h1)
    print(h2)
    print(sep)

    for nodo in nodos:
        line = f"| {timestamp.ljust(ts_width)} | {nodo.ljust(nodo_w)}"

        for cluster in cluster_order:
            cur_map = current_data.get(cluster, {})
            prev_map = previous_data.get(cluster, {})

            r = cur_map.get(nodo)
            cur_val = r.encolados if r else 0
            ac_val = r.active_recips if r else 0
            prev_val = prev_map.get(nodo, cur_val)
            var = trend_symbol(prev_val, cur_val)
            has_error = bool(r.error) if r else False

            chunk = fmt_chunk(prev_val, cur_val, var, ac_val, cluster, has_error)
            line += f"{cluster_sep_colored}{chunk}"

        line += " |"
        print(line)

    print(sep)

    total_line = f"| {'':<{ts_width}} | {'TOTAL'.ljust(nodo_w)}"

    for cluster in cluster_order:
        cur_map = current_data.get(cluster, {})
        prev_map = previous_data.get(cluster, {})

        total_cur = sum(r.encolados for r in cur_map.values())
        total_prev = sum(
            prev_map.get(n, cur_map.get(n).encolados if cur_map.get(n) else 0)
            for n in nodos
        )
        total_ac = sum(r.active_recips for r in cur_map.values())
        var = trend_symbol(total_prev, total_cur)

        ant_txt = fmt_num_dot(total_prev).rjust(ant_w)
        q_txt = queue_color(cluster, fmt_num_dot(total_cur).rjust(act_w), total_cur)
        var_txt = var.center(var_w)
        ac_txt = fmt_num_dot(total_ac).rjust(ac_w)

        chunk = f"{ant_txt}{col_sep}{q_txt}{col_sep}{var_txt}{col_sep}{ac_txt}"
        total_line += f"{cluster_sep_colored}{chunk}"

    total_line += " |"
    print(total_line)

            ADV_THRESHOLDS["conn_critical"],

            status_txt = f"{RED}{status_txt}{RESET}"
            cpu_txt = f"{RED}{str(r.cpu).rjust(cpu_w)}{RESET}"
            ram_txt = f"{RED}{str(r.ram).rjust(ram_w)}{RESET}"

    h2 = (
        f"| {'':<{nodo_w}}"
        f"{cluster_sep_colored}{sub}"
        f"{cluster_sep_colored}{sub}"
        f"{cluster_sep_colored}{sub}"
        f"{cluster_sep_colored}{sub} |"
    )

    sep = "-" * (
        2 + nodo_w +
        (len(cluster_sep_plain) + cluster_chunk_width) * 4 + 2
    )

    print(h1)
    print(h2)
    print(sep)

    for nodo in nodos:
        line = f"| {nodo.ljust(nodo_w)}"

        for cluster in cluster_order:
            r = current_data.get(cluster, {}).get(nodo)
            prev_q = previous_data.get(cluster, {}).get(nodo, r.encolados if r else 0)
            chunk = fmt_chunk(r, prev_q)
            line += f"{cluster_sep_colored}{chunk}"

        line += " |"
        print(line)


def print_status_incidents(current_data: Dict[str, Dict[str, Result]]) -> None:
    incidents = build_status_incidents(current_data)

    print("INCIDENCIAS DE STATUS\n")

    if not incidents:
        print(f"{GREEN}OK - Sin incidencias de status{RESET}")
        return

    for cluster, nodo, incidencia in incidents:
        color = RED if incidencia == "Sin respuesta" else YELLOW
        print(f"{color}{cluster} {nodo} → {incidencia}{RESET}")

def run_monitor(auth: Tuple[str, str]) -> None:
    previous_data: Dict[str, Dict[str, int]] = {}

    listener = threading.Thread(target=esc_listener, daemon=True)
    listener.start()

    with requests.Session() as session:
        while not stop_event.is_set():
            timestamp = datetime.now().strftime("%H:%S")
            current_data: Dict[str, Dict[str, Result]] = {}

            with ThreadPoolExecutor(max_workers=len(GROUPS)) as ex:
                fut_map = {
                    ex.submit(fetch_cluster, session, g, auth, timestamp): g["cluster"]
                    for g in GROUPS
                }

                for fut in as_completed(fut_map):
                    cluster = fut_map[fut]
                    results = fut.result()
                    current_data[cluster] = {r.nodo: r for r in results}

            clear_screen()
            print("MONITOR COLAS ESA\n")
            print_cluster_summary_with_side_panels(current_data, previous_data)
            print()
            print_orange_separator()
            print()
            print_queue_table(timestamp, current_data, previous_data)
            print()
            print_orange_separator()
            print()
            print_node_health_table(current_data, previous_data)
            print()
            print_orange_separator()
            print()
            print_status_incidents(current_data)

            print("\nPulsa ESC para salir")

            previous_data = {
                c: {n: r.encolados for n, r in m.items()}
                for c, m in current_data.items()
            }

            for _ in range(REFRESH_SECONDS * 10):
                if stop_event.is_set():
                    break
                time.sleep(0.1)

    clear_screen()
    print("Monitor detenido (ESC pulsado)")


def main() -> None:
    creds = load_env_file(CREDENTIALS_FILE)
    auth = (creds["GLOBAL_USERNAME"], creds["GLOBAL_PASSWORD"])

    clear_screen()
    run_monitor(auth)


if __name__ == "__main__":
    main()
