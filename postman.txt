#!/usr/bin/env python3
# -*- coding: utf-8 -*-

import os
import re
import time
import threading
from dataclasses import dataclass
from typing import Optional, List, Dict, Tuple
from datetime import datetime

import requests
import urllib3
import xml.etree.ElementTree as ET
from concurrent.futures import ThreadPoolExecutor, as_completed


# =========================
# Config
# =========================
CREDENTIALS_FILE = "credentials.env"

TIMEOUT_SECONDS = 12
MAX_WORKERS = 12
RETRIES = 2

REFRESH_SECONDS = 60

# Umbrales por nodo
WARNING_THRESHOLD = 50
CRITICAL_THRESHOLD = 100

# Umbrales para TOTAL
TOTAL_WARNING_THRESHOLD = 200
TOTAL_CRITICAL_THRESHOLD = 300

VERIFY_SSL = False
if not VERIFY_SSL:
    urllib3.disable_warnings(urllib3.exceptions.InsecureRequestWarning)

RED = "\033[91m"
YELLOW = "\033[93m"
RESET = "\033[0m"

GROUPS = [
    {"cluster": "Europa",       "pattern": "https://dh529-esa{idx}.c3s2.iphmx.com/xml/status", "start": 1, "end": 20},
    {"cluster": "América",      "pattern": "https://dh5504-esa{idx}.c3s2.iphmx.com/xml/status", "start": 1, "end": 20},
    {"cluster": "Brasil",       "pattern": "https://dh6121-esa{idx}.iphmx.com/xml/status",      "start": 1, "end": 20},
    {"cluster": "Aplicaciones", "pattern": "https://dh5505-esa{idx}.c3s2.iphmx.com/xml/status", "start": 1, "end": 20},
]


# =========================
# Model
# =========================
@dataclass
class Result:
    timestamp: str
    nodo: str
    cluster: str
    status: str
    encolados: int
    error: Optional[str] = None


# =========================
# Helpers
# =========================
def script_dir() -> str:
    return os.path.dirname(os.path.abspath(__file__))


def load_env_file(path: str) -> Dict[str, str]:
    data: Dict[str, str] = {}
    if not os.path.exists(path):
        raise FileNotFoundError(f"No existe el archivo de credenciales: {path}")

    with open(path, "r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line or line.startswith("#") or "=" not in line:
                continue
            k, v = line.split("=", 1)
            data[k.strip()] = v.strip()
    return data


def nodo_from_index(idx: int) -> str:
    return f"ESA{idx:02d}"


def to_int(val: Optional[str]) -> int:
    if val is None:
        return 0
    s = str(val).strip()
    if not s or s.upper() == "N/A":
        return 0
    m = re.search(r"(\d+(\.\d+)?)", s)
    if not m:
        return 0
    try:
        return int(float(m.group(1)))
    except Exception:
        return 0


def fmt_num_dot(n: int) -> str:
    return f"{n:,}".replace(",", ".")


def trend_symbol(prev: int, cur: int) -> str:
    if cur > prev:
        return "+"
    if cur < prev:
        return "-"
    return "="


def clear_screen() -> None:
    os.system("cls" if os.name == "nt" else "clear")


def colorize_node(text: str, current_value: int, has_error: bool = False) -> str:
    if has_error:
        return text
    if current_value >= CRITICAL_THRESHOLD:
        return f"{RED}{text}{RESET}"
    if current_value >= WARNING_THRESHOLD:
        return f"{YELLOW}{text}{RESET}"
    return text


def colorize_total(text: str, total_value: int) -> str:
    if total_value >= TOTAL_CRITICAL_THRESHOLD:
        return f"{RED}{text}{RESET}"
    if total_value >= TOTAL_WARNING_THRESHOLD:
        return f"{YELLOW}{text}{RESET}"
    return text


# =========================
# XML parsing
# =========================
def parse_encolados(xml_text: str) -> Tuple[str, int]:
    root = ET.fromstring(xml_text)

    system_status = "No disponible"
    sys_node = root.find(".//system")
    if sys_node is not None:
        system_status = sys_node.attrib.get("status", system_status)

    encolados = 0
    for gauge in root.findall(".//gauge"):
        if gauge.attrib.get("name") == "msgs_in_work_queue":
            val = gauge.attrib.get("current")
            if val is None:
                cur = gauge.find("./current")
                if cur is not None and cur.text:
                    val = cur.text.strip()
            encolados = to_int(val)
            break

    return system_status, encolados


# =========================
# Network
# =========================
def fetch_one(
    session: requests.Session,
    url: str,
    auth: Tuple[str, str],
    timestamp: str,
    cluster: str,
    idx: int,
) -> Result:
    nodo = nodo_from_index(idx)
    headers = {"Accept": "application/xml"}

    last_err: Optional[str] = None
    for _ in range(RETRIES + 1):
        try:
            r = session.get(
                url,
                headers=headers,
                auth=auth,
                timeout=TIMEOUT_SECONDS,
                verify=VERIFY_SSL,
            )

            if r.status_code != 200:
                return Result(timestamp, nodo, cluster, "Error conexión", 0, f"HTTP {r.status_code}")

            status, encolados = parse_encolados(r.text)
            return Result(timestamp, nodo, cluster, status, encolados, None)

        except requests.exceptions.RequestException as e:
            last_err = str(e)

    return Result(timestamp, nodo, cluster, "Error conexión", 0, last_err or "RequestException")


def fetch_cluster(
    session: requests.Session,
    group: Dict,
    auth: Tuple[str, str],
    timestamp: str,
) -> List[Result]:
    cluster = group["cluster"]
    pattern = group["pattern"]
    start = int(group["start"])
    end = int(group["end"])

    targets = [(i, pattern.format(idx=i)) for i in range(start, end + 1)]

    results: List[Result] = []
    with ThreadPoolExecutor(max_workers=MAX_WORKERS) as ex:
        futures = [
            ex.submit(fetch_one, session, url, auth, timestamp, cluster, idx)
            for idx, url in targets
        ]
        for fut in as_completed(futures):
            results.append(fut.result())

    return sorted(results, key=lambda r: r.nodo)


# =========================
# Table
# =========================
def print_combined_table(
    timestamp: str,
    current_data: Dict[str, Dict[str, Result]],
    previous_data: Dict[str, Dict[str, int]],
) -> None:
    cluster_order = ["Europa", "América", "Brasil", "Aplicaciones"]
    nodos = [f"ESA{i:02d}" for i in range(1, 21)]

    ts_width = max(len("Timestamp"), len(timestamp))
    nodo_width = len("Nodo") + 1

    ant_w = 12
    act_w = 12
    var_w = 5

    col_sep = " | "
    cluster_sep = " || "

    cluster_chunk_width = ant_w + len(col_sep) + act_w + len(col_sep) + var_w

    def fmt_chunk(prev_val: int, cur_val: int, var_val: str) -> str:
        return (
            f"{fmt_num_dot(prev_val).rjust(ant_w)}{col_sep}"
            f"{fmt_num_dot(cur_val).rjust(act_w)}{col_sep}"
            f"{var_val.center(var_w)}"
        )

    h1 = (
        f"| {'Timestamp'.ljust(ts_width)}{col_sep}"
        f"{'Nodo'.ljust(nodo_width)}{cluster_sep}"
        f"{'Europa'.center(cluster_chunk_width)}{cluster_sep}"
        f"{'América'.center(cluster_chunk_width)}{cluster_sep}"
        f"{'Brasil'.center(cluster_chunk_width)}{cluster_sep}"
        f"{'Aplicaciones'.center(cluster_chunk_width)} |"
    )

    sub = (
        f"{'Ant'.center(ant_w)}{col_sep}"
        f"{'Act'.center(act_w)}{col_sep}"
        f"{'Var'.center(var_w)}"
    )

    h2 = (
        f"| {''.ljust(ts_width)}{col_sep}"
        f"{''.ljust(nodo_width)}{cluster_sep}"
        f"{sub}{cluster_sep}{sub}{cluster_sep}{sub}{cluster_sep}{sub} |"
    )

    sep = "-" * max(len(h1), len(h2))

    print("MONITOR GLOBAL ESA | refresco cada 60s | pulsa ESC para salir\n")
    print(h1)
    print(h2)
    print(sep)

    for nodo in nodos:
        line = f"| {timestamp.ljust(ts_width)}{col_sep}{nodo.ljust(nodo_width)}"

        for cluster in cluster_order:
            cur_map = current_data.get(cluster, {})
            prev_map = previous_data.get(cluster, {})

            cur_res = cur_map.get(nodo)
            cur_val = cur_res.encolados if cur_res else 0
            prev_val = prev_map.get(nodo, cur_val)
            var_val = trend_symbol(prev_val, cur_val)
            has_error = bool(cur_res.error) if cur_res else False

            chunk = fmt_chunk(prev_val, cur_val, var_val)
            chunk = colorize_node(chunk, cur_val, has_error=has_error)
            line += f"{cluster_sep}{chunk}"

        line += " |"
        print(line)

    print(sep)

    total_line = f"| {''.ljust(ts_width)}{col_sep}{'TOTAL'.ljust(nodo_width)}"

    for cluster in cluster_order:
        cur_map = current_data.get(cluster, {})
        prev_map = previous_data.get(cluster, {})

        total_cur = sum(r.encolados for r in cur_map.values())
        total_prev = sum(
            prev_map.get(nodo, cur_map.get(nodo).encolados if cur_map.get(nodo) else 0)
            for nodo in nodos
        )
        total_var = trend_symbol(total_prev, total_cur)

        chunk = fmt_chunk(total_prev, total_cur, total_var)
        chunk = colorize_total(chunk, total_cur)
        total_line += f"{cluster_sep}{chunk}"

    total_line += " |"
    print(total_line)


def print_alerts(current_data: Dict[str, Dict[str, Result]]) -> None:
    alerts = []

    for cluster, nodo_map in current_data.items():
        for nodo, r in nodo_map.items():
            if r.error:
                continue
            if r.encolados >= CRITICAL_THRESHOLD:
                alerts.append(("CRITICAL", cluster, nodo, r.encolados))
            elif r.encolados >= WARNING_THRESHOLD:
                alerts.append(("WARNING", cluster, nodo, r.encolados))

    if not alerts:
        return

    critical_count = sum(1 for a in alerts if a[0] == "CRITICAL")
    warning_count = sum(1 for a in alerts if a[0] == "WARNING")

    print(f"\nALERTAS - CRITICAL: {critical_count} | WARNING: {warning_count}")

    for level, cluster, nodo, value in sorted(alerts, key=lambda x: x[3], reverse=True):
        if level == "CRITICAL":
            print(f"{RED}CRITICAL{RESET} - {cluster} {nodo} -> encolados={fmt_num_dot(value)}")
        else:
            print(f"{YELLOW}WARNING{RESET}  - {cluster} {nodo} -> encolados={fmt_num_dot(value)}")


# =========================
# ESC listener
# =========================
def start_esc_listener(stop_event: threading.Event) -> None:
    try:
        import msvcrt  # type: ignore

        def _win_loop():
            while not stop_event.is_set():
                if msvcrt.kbhit():
                    ch = msvcrt.getch()
                    if ch == b"\x1b":
                        stop_event.set()
                        return
                time.sleep(0.05)

        threading.Thread(target=_win_loop, daemon=True).start()
        return

    except Exception:
        pass

    import sys
    import select
    import termios
    import tty

    def _posix_loop():
        fd = sys.stdin.fileno()
        old = termios.tcgetattr(fd)
        try:
            tty.setcbreak(fd)
            while not stop_event.is_set():
                r, _, _ = select.select([sys.stdin], [], [], 0.1)
                if r:
                    ch = sys.stdin.read(1)
                    if ch == "\x1b":
                        stop_event.set()
                        return
        finally:
            termios.tcsetattr(fd, termios.TCSADRAIN, old)

    threading.Thread(target=_posix_loop, daemon=True).start()


# =========================
# Main loop
# =========================
def run_monitor(auth: Tuple[str, str]) -> None:
    stop_event = threading.Event()
    start_esc_listener(stop_event)

    previous_data: Dict[str, Dict[str, int]] = {}

    with requests.Session() as session:
        while not stop_event.is_set():
            timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
            current_data: Dict[str, Dict[str, Result]] = {}

            with ThreadPoolExecutor(max_workers=len(GROUPS)) as ex:
                fut_map = {
                    ex.submit(fetch_cluster, session, group, auth, timestamp): group["cluster"]
                    for group in GROUPS
                }
                for fut in as_completed(fut_map):
                    cluster = fut_map[fut]
                    cluster_results = fut.result()
                    current_data[cluster] = {r.nodo: r for r in cluster_results}

            clear_screen()
            print_combined_table(timestamp, current_data, previous_data)
            print_alerts(current_data)
            print("\n(Refrescando... pulsa ESC para salir)")

            previous_data = {
                cluster: {nodo: r.encolados for nodo, r in nodo_map.items()}
                for cluster, nodo_map in current_data.items()
            }

            for _ in range(REFRESH_SECONDS * 10):
                if stop_event.is_set():
                    break
                time.sleep(0.1)

    clear_screen()
    print("Salida solicitada. Monitor detenido.")


def main() -> int:
    base = script_dir()
    creds = load_env_file(os.path.join(base, CREDENTIALS_FILE))

    username = creds.get("GLOBAL_USERNAME")
    password = creds.get("GLOBAL_PASSWORD")
    if not username or not password:
        print("Faltan GLOBAL_USERNAME y/o GLOBAL_PASSWORD en credentials.env")
        return 2

    clear_screen()
    run_monitor((username, password))
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
