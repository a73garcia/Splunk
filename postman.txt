#!/usr/bin/env python3
# -*- coding: utf-8 -*-

import os
import time
import threading
import msvcrt
from dataclasses import dataclass
from typing import Optional, List, Dict
from datetime import datetime
import xml.etree.ElementTree as ET
from concurrent.futures import ThreadPoolExecutor, as_completed

import requests
import urllib3


CREDENTIALS_FILE = "credentials.env"

TIMEOUT_SECONDS = 12
MAX_WORKERS = 12
REFRESH_SECONDS = 60

VERIFY_SSL = False
if not VERIFY_SSL:
    urllib3.disable_warnings(urllib3.exceptions.InsecureRequestWarning)

RED = "\033[91m"
YELLOW = "\033[93m"
GREEN = "\033[92m"
ORANGE = "\033[38;5;208m"
RESET = "\033[0m"


GROUPS = [
    {"cluster": "Europa", "pattern": "https://dh529-esa{idx}.c3s2.iphmx.com/xml/status"},
    {"cluster": "América", "pattern": "https://dh5504-esa{idx}.c3s2.iphmx.com/xml/status"},
    {"cluster": "Brasil", "pattern": "https://dh6121-esa{idx}.iphmx.com/xml/status"},
    {"cluster": "Aplicaciones", "pattern": "https://dh5505-esa{idx}.c3s2.iphmx.com/xml/status"},
]

stop_event = threading.Event()


@dataclass
class Result:
    nodo: str
    cluster: str
    status: str
    encolados: int
    cpu: int
    ram: int
    conn_in: int
    conn_out: int
    active_recips: int
    error: Optional[str] = None


# ---------------------------------------------------------
# helpers
# ---------------------------------------------------------

def clear_screen():
    os.system("cls")


def orange_cluster_sep():
    return f"{ORANGE}||{RESET}"


def fmt(n):
    return f"{n:,}".replace(",", ".")


def load_credentials():

    creds = {}

    with open(CREDENTIALS_FILE) as f:

        for line in f:

            if "=" in line:

                k, v = line.strip().split("=", 1)

                creds[k] = v

    return creds


# ---------------------------------------------------------
# XML parsing
# ---------------------------------------------------------

def parse_status(xml):

    root = ET.fromstring(xml)

    status = "unknown"

    sys = root.find(".//system")

    if sys is not None:
        status = sys.attrib.get("status", status)

    gauges = {
        "msgs_in_work_queue": 0,
        "cpu_utilization": 0,
        "ram_utilization": 0,
        "conn_in": 0,
        "conn_out": 0,
        "active_recips": 0,
    }

    for g in root.findall(".//gauge"):

        name = g.attrib.get("name")

        if name in gauges:

            try:
                gauges[name] = int(float(g.attrib.get("current")))
            except:
                gauges[name] = 0

    return status, gauges

def fetch_node(session, url, auth, cluster, idx):

    nodo = f"ESA{idx:02d}"

    try:

        r = session.get(url, auth=auth, timeout=TIMEOUT_SECONDS, verify=False)

        if r.status_code != 200:
            return Result(nodo, cluster, "offline", 0,0,0,0,0,0,"HTTP")

        status, m = parse_status(r.text)

        return Result(
            nodo,
            cluster,
            status,
            m["msgs_in_work_queue"],
            m["cpu_utilization"],
            m["ram_utilization"],
            m["conn_in"],
            m["conn_out"],
            m["active_recips"],
        )

    except Exception:

        return Result(nodo, cluster, "offline",0,0,0,0,0,0,"ERROR")


def fetch_cluster(session, group, auth):

    cluster = group["cluster"]
    pattern = group["pattern"]

    results = []

    with ThreadPoolExecutor(max_workers=MAX_WORKERS) as ex:

        futures = []

        for i in range(1,21):

            url = pattern.format(idx=i)

            futures.append(ex.submit(fetch_node, session, url, auth, cluster, i))

        for f in as_completed(futures):

            results.append(f.result())

    return results

def detect_queue_storm(data, prev):

    findings = []

    for cluster,nodes in data.items():

        for nodo,r in nodes.items():

            prev_val = prev.get(cluster,{}).get(nodo,r.encolados)

            growth = r.encolados - prev_val

            if growth > 50:

                findings.append(f"{cluster} {nodo}: cola creciendo +{fmt(growth)}")

    return findings


def detect_smtp_saturation(data):

    findings = []

    for cluster,nodes in data.items():

        for nodo,r in nodes.items():

            if r.conn_out > 40:

                findings.append(f"{cluster} {nodo}: salida SMTP alta ({r.conn_out})")

    return findings


def detect_cluster_imbalance(data):

    findings = []

    for cluster,nodes in data.items():

        queues = [r.encolados for r in nodes.values()]

        if not queues:
            continue

        max_q = max(queues)
        avg_q = sum(queues)/len(queues)

        if avg_q > 0 and max_q/avg_q > 2:

            findings.append(f"{cluster}: cluster desbalanceado")

    return findings

# panel superior

def build_node_alert_lines(data):

    lines=["ALERTAS NODOS",""]

    for cluster,nodes in data.items():

        for nodo,r in nodes.items():

            if r.encolados>100:

                lines.append(f"{RED}{cluster} {nodo} cola={fmt(r.encolados)}{RESET}")

    if len(lines)==2:

        lines.append(f"{GREEN}Sin alertas{RESET}")

    return lines


def build_risk_nodes_lines(data,prev):

    lines=["NODOS EN RIESGO",""]

    for cluster,nodes in data.items():

        for nodo,r in nodes.items():

            prev_val=prev.get(cluster,{}).get(nodo,r.encolados)

            if r.encolados-prev_val>20:

                lines.append(f"{YELLOW}{cluster} {nodo} creciendo{RESET}")

    if len(lines)==2:

        lines.append(f"{GREEN}Sin nodos en riesgo{RESET}")

    return lines


def print_two_column_panels(left,right,width=70):

    top=f"{ORANGE}╔{'═'*width}╗{RESET}"

    print(top+"   "+top)

    rows=max(len(left),len(right))

    left+=[""]*(rows-len(left))
    right+=[""]*(rows-len(right))

    for l,r in zip(left,right):

        print(f"{ORANGE}║{RESET}{l.ljust(width)}{ORANGE}║{RESET}   {ORANGE}║{RESET}{r.ljust(width)}{ORANGE}║{RESET}")

    bottom=f"{ORANGE}╚{'═'*width}╝{RESET}"

    print(bottom+"   "+bottom)


def print_cluster_summary_with_side_panels(data,prev):

    left=["RESUMEN CLUSTERS",""]

    for cluster,nodes in data.items():

        total=sum(r.encolados for r in nodes.values())

        left.append(f"{cluster:<14} {fmt(total):>10}")

    alerts=build_node_alert_lines(data)

    risks=build_risk_nodes_lines(data,prev)

    right=alerts+[""]+risks

    print_two_column_panels(left,right)

def print_queue_table(ts,data,prev):

    cluster_sep=f" {orange_cluster_sep()} "

    print("ESTADO ENCOLADOS NODOS\n")

    for i in range(1,21):

        nodo=f"ESA{i:02d}"

        row=f"{ts} {nodo}"

        for cluster in ["Europa","América","Brasil","Aplicaciones"]:

            r=data.get(cluster,{}).get(nodo)

            if r:

                prev_val=prev.get(cluster,{}).get(nodo,r.encolados)

                diff=r.encolados-prev_val

                sym="↑" if diff>0 else "↓" if diff<0 else "→"

                row+=f"{cluster_sep}{fmt(prev_val)} {fmt(r.encolados)} {sym} {fmt(r.active_recips)}"

        print(row)


def print_health_table(data):

    cluster_sep=f" {orange_cluster_sep()} "

    print("\nSALUD OPERATIVA\n")

    for i in range(1,21):

        nodo=f"ESA{i:02d}"

        row=nodo

        for cluster in ["Europa","América","Brasil","Aplicaciones"]:

            r=data.get(cluster,{}).get(nodo)

            if r:

                status="online"

                if r.status.lower()!="online" or r.error:

                    status="offline"

                row+=f"{cluster_sep}{status} {r.cpu} {r.ram} {r.conn_out}"

        print(row)

def print_status_incidents(data):

    print("\nINCIDENCIAS\n")

    for cluster,nodes in data.items():

        for nodo,r in nodes.items():

            if r.error:

                print(f"{RED}{cluster} {nodo} sin respuesta{RESET}")

            elif r.status.lower()!="online":

                print(f"{YELLOW}{cluster} {nodo} status={r.status}{RESET}")


# ---------------------------------------------------------
# análisis avanzado
# ---------------------------------------------------------

def print_advanced_findings(data,prev):

    print("\nANÁLISIS AVANZADO\n")

    findings=[]

    findings+=detect_queue_storm(data,prev)
    findings+=detect_smtp_saturation(data)
    findings+=detect_cluster_imbalance(data)

    if not findings:

        print(f"{GREEN}Sistema estable sin anomalías detectadas{RESET}")
        return

    for f in findings:

        print(f"{YELLOW}{f}{RESET}")


# ---------------------------------------------------------
# ESC listener
# ---------------------------------------------------------

def esc_listener():

    while not stop_event.is_set():

        if msvcrt.kbhit():

            if msvcrt.getch()==b"\x1b":

                stop_event.set()

        time.sleep(0.1)


# ---------------------------------------------------------
# monitor
# ---------------------------------------------------------

def run_monitor(auth):

    prev={}

    listener=threading.Thread(target=esc_listener,daemon=True)

    listener.start()

    with requests.Session() as session:

        while not stop_event.is_set():

            ts=datetime.now().strftime("%H:%M")

            data={}

            with ThreadPoolExecutor(max_workers=4) as ex:

                futures={ex.submit(fetch_cluster,session,g,auth):g["cluster"] for g in GROUPS}

                for f in as_completed(futures):

                    cluster=futures[f]

                    results=f.result()

                    data[cluster]={r.nodo:r for r in results}

            clear_screen()

            print("MONITOR ESA\n")

            print_cluster_summary_with_side_panels(data,prev)

            print()

            print_queue_table(ts,data,prev)

            print_health_table(data)

            print_status_incidents(data)

            print_advanced_findings(data,prev)

            prev={c:{n:r.encolados for n,r in m.items()} for c,m in data.items()}

            for _ in range(REFRESH_SECONDS*10):

                if stop_event.is_set():
                    break

                time.sleep(0.1)

    clear_screen()

    print("Monitor detenido")


# ---------------------------------------------------------
# main
# ---------------------------------------------------------

def main():

    creds=load_credentials()

    auth=(creds["GLOBAL_USERNAME"],creds["GLOBAL_PASSWORD"])

    clear_screen()

    run_monitor(auth)


if __name__=="__main__":

    main()
