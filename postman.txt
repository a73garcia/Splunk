#!/usr/bin/env python3
# -*- coding: utf-8 -*-

import os
import csv
from dataclasses import dataclass
from typing import Optional, List, Dict, Tuple
from datetime import datetime

import requests
import urllib3
import xml.etree.ElementTree as ET
from concurrent.futures import ThreadPoolExecutor, as_completed


# -------------------------
# Config
# -------------------------
CREDENTIALS_FILE = "credentials.env"
OUTPUT_FOLDER = "encolados"

TIMEOUT_SECONDS = 12
MAX_WORKERS = 12
RETRIES = 2

WARNING_THRESHOLD = 50
CRITICAL_THRESHOLD = 100

VERIFY_SSL = False
if not VERIFY_SSL:
    urllib3.disable_warnings(urllib3.exceptions.InsecureRequestWarning)

RED = "\033[91m"
YELLOW = "\033[93m"
RESET = "\033[0m"


GROUPS = [
    {"cluster": "Europa",       "pattern": "https://dh529-esa{idx}.c3s2.iphmx.com/xml/status", "start": 1, "end": 20},
    {"cluster": "Aplicaciones", "pattern": "https://dh5505-esa{idx}.c3s2.iphmx.com/xml/status", "start": 1, "end": 20},
    {"cluster": "América",      "pattern": "https://dh5504-esa{idx}.c3s2.iphmx.com/xml/status", "start": 1, "end": 20},
    {"cluster": "Brasil",       "pattern": "https://dh6121-esa{idx}.iphmx.com/xml/status",      "start": 1, "end": 20},
]


@dataclass
class Result:
    timestamp: str
    nodo: str
    cluster: str
    status: str
    gauge: str
    encolados: int
    error: Optional[str] = None


def script_dir():
    return os.path.dirname(os.path.abspath(__file__))


def load_env_file(path):
    data = {}
    with open(path) as f:
        for line in f:
            if "=" in line:
                k, v = line.strip().split("=", 1)
                data[k] = v
    return data


def nodo_from_index(idx):
    return f"ESA{idx:02d}"


def colorize(text, value):
    if value >= CRITICAL_THRESHOLD:
        return f"{RED}{text}{RESET}"
    if value >= WARNING_THRESHOLD:
        return f"{YELLOW}{text}{RESET}"
    return text


def parse_encolados(xml_text):
    root = ET.fromstring(xml_text)

    system_status = "No disponible"
    sys_node = root.find(".//system")
    if sys_node is not None:
        system_status = sys_node.attrib.get("status", system_status)

    for gauge in root.findall(".//gauge"):
        if gauge.attrib.get("name") == "msgs_in_work_queue":
            val = gauge.attrib.get("current")

            if val is None:
                current_node = gauge.find("./current")
                if current_node is not None and current_node.text:
                    val = current_node.text.strip()

            if val is None or val.upper() == "N/A":
                return system_status, 0

            return system_status, int(val)

    return system_status, 0


def fetch_one(session, url, auth, timestamp, cluster, idx):

    node = nodo_from_index(idx)

    try:
        r = session.get(
            url,
            auth=auth,
            timeout=TIMEOUT_SECONDS,
            verify=VERIFY_SSL,
        )

        if r.status_code != 200:
            return Result(timestamp, node, cluster, "Error", "msgs_in_work_queue", 0, f"HTTP {r.status_code}")

        status, q = parse_encolados(r.text)
        return Result(timestamp, node, cluster, status, "msgs_in_work_queue", q)

    except Exception as e:
        return Result(timestamp, node, cluster, "Error conexión", "msgs_in_work_queue", 0, str(e))


def append_csv(results, path):

    file_exists = os.path.exists(path)

    with open(path, "a", newline="", encoding="utf-8") as f:

        writer = csv.writer(f)

        if not file_exists:
            writer.writerow(["timestamp","nodo","cluster","status","gauge","encolados","error"])

        for r in results:
            writer.writerow([
                r.timestamp,
                r.nodo,
                r.cluster,
                r.status,
                r.gauge,
                r.encolados,
                r.error or ""
            ])


def print_table(cluster, results):

    results = sorted(results, key=lambda r: r.encolados, reverse=True)

    print(f"\n=== {cluster} ===")
    print("Nodo | Encolados")
    print("----------------")

    for r in results:
        line = f"{r.nodo:5} | {r.encolados}"
        print(colorize(line, r.encolados))


def print_summary(all_results):

    print("\n============== RESUMEN CLUSTERS ==============")

    headers = ["Cluster","Total Encolados","Nodo","Encolados","Errores"]

    rows = []
    rows_with_sep = []

    for cluster, results in all_results.items():

        total = sum(r.encolados for r in results)
        errors = sum(1 for r in results if r.error)

        sorted_nodes = sorted(results, key=lambda r: r.encolados, reverse=True)

        for i in range(3):

            if i < len(sorted_nodes):
                r = sorted_nodes[i]
                row = [
                    cluster if i == 0 else "",
                    str(total) if i == 0 else "",
                    r.nodo,
                    str(r.encolados),
                    str(errors) if i == 0 else ""
                ]
            else:
                row = [
                    cluster if i == 0 else "",
                    str(total) if i == 0 else "",
                    "-",
                    "0",
                    str(errors) if i == 0 else ""
                ]

            rows.append(row)
            rows_with_sep.append(("ROW", row))

        rows_with_sep.append(("SEP", None))

    widths = [len(h) for h in headers]

    for row in rows:
        for i, cell in enumerate(row):
            widths[i] = max(widths[i], len(cell))

    def fmt(vals):
        return " | ".join(vals[i].ljust(widths[i]) for i in range(len(vals)))

    sep = "-+-".join("-" * w for w in widths)
    cluster_sep = "=" * len(sep)

    print(fmt(headers))
    print(sep)

    for kind, row in rows_with_sep:

        if kind == "SEP":
            print(cluster_sep)
            continue

        encolados = int(row[3])
        line = fmt(row)

        print(colorize(line, encolados))


def main():

    now = datetime.now()
    timestamp = now.strftime("%Y-%m-%d %H:%M:%S")

    base = script_dir()
    out_dir = os.path.join(base, OUTPUT_FOLDER)
    os.makedirs(out_dir, exist_ok=True)

    creds = load_env_file(os.path.join(base, CREDENTIALS_FILE))

    auth = (
        creds["GLOBAL_USERNAME"],
        creds["GLOBAL_PASSWORD"]
    )

    all_results = {}

    with requests.Session() as session:

        for g in GROUPS:

            cluster = g["cluster"]

            targets = [
                (i, g["pattern"].format(idx=i))
                for i in range(g["start"], g["end"] + 1)
            ]

            results = []

            with ThreadPoolExecutor(max_workers=MAX_WORKERS) as ex:

                futures = [
                    ex.submit(fetch_one, session, url, auth, timestamp, cluster, idx)
                    for idx, url in targets
                ]

                for fut in as_completed(futures):
                    results.append(fut.result())

            all_results[cluster] = results

            print_table(cluster, results)

            csv_path = os.path.join(
                out_dir,
                f"encolados_{cluster}_{now.strftime('%Y-%m')}.csv"
            )

            append_csv(results, csv_path)

    print_summary(all_results)


if __name__ == "__main__":
    main()
