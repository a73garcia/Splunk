#!/usr/bin/env python3
# -*- coding: utf-8 -*-

import os
import csv
from dataclasses import dataclass
from typing import Optional, List
from datetime import datetime

import requests
import urllib3
import xml.etree.ElementTree as ET
from concurrent.futures import ThreadPoolExecutor, as_completed


CREDENTIALS_FILE = "credentials.env"
OUTPUT_FOLDER = "encolados"

TIMEOUT_SECONDS = 12
MAX_WORKERS = 10

VERIFY_SSL = False

if not VERIFY_SSL:
    urllib3.disable_warnings(urllib3.exceptions.InsecureRequestWarning)


GROUPS = [

    {"cluster": "Europa", "prefix": "dh529", "start": 1, "end": 20},

    {"cluster": "Aplicaciones", "prefix": "dh5505", "start": 1, "end": 20},

    {"cluster": "América", "prefix": "dh5504", "start": 1, "end": 20},

    {"cluster": "Brasil", "prefix": "dh6121", "start": 1, "end": 20},

]


@dataclass
class Result:
    timestamp: str
    nodo: str
    cluster: str
    status: str
    encolados: int
    error: Optional[str] = None


def script_dir():
    return os.path.dirname(os.path.abspath(__file__))


def load_env_file(path):

    data = {}

    with open(path) as f:

        for line in f:

            line = line.strip()

            if not line or line.startswith("#"):
                continue

            k, v = line.split("=", 1)

            data[k.strip()] = v.strip()

    return data


def nodo(idx):

    return f"ESA{idx:02d}"


def build_url(prefix, idx):

    return f"https://{prefix}-esa{idx}.c3s2.iphmx.com/xml/status"


def parse_queue(xml_text):

    root = ET.fromstring(xml_text)

    status = "No disponible"

    sys_node = root.find(".//system")

    if sys_node is not None:
        status = sys_node.attrib.get("status", status)

    for gauge in root.findall(".//gauge"):

        if gauge.attrib.get("name") == "msgs_in_work_queue":

            value = gauge.attrib.get("current")

            if value is None or value.upper() == "N/A":
                return status, 0

            try:
                return status, int(value)
            except:
                return status, 0

    return status, 0


def fetch(session, prefix, idx, auth, timestamp, cluster):

    node_name = nodo(idx)

    url = build_url(prefix, idx)

    try:

        r = session.get(
            url,
            auth=auth,
            timeout=TIMEOUT_SECONDS,
            verify=VERIFY_SSL
        )

        if r.status_code != 200:

            return Result(timestamp, node_name, cluster, "Error conexión", 0)

        status, queue = parse_queue(r.text)

        return Result(timestamp, node_name, cluster, status, queue)

    except Exception as e:

        return Result(timestamp, node_name, cluster, "Error conexión", 0, str(e))


def print_table(results):

    headers = ["Timestamp", "Nodo", "Cluster", "Status", "Encolados", "Error"]

    print(" | ".join(headers))
    print("-" * 80)

    for r in results:

        print(
            f"{r.timestamp} | {r.nodo} | {r.cluster} | {r.status} | {r.encolados} | {r.error or ''}"
        )


def save_csv(results, path):

    with open(path, "w", newline="", encoding="utf-8") as f:

        writer = csv.writer(f)

        writer.writerow(["timestamp", "nodo", "cluster", "status", "encolados", "error"])

        for r in results:

            writer.writerow([

                r.timestamp,
                r.nodo,
                r.cluster,
                r.status,
                r.encolados,
                r.error or ""

            ])


def main():

    now = datetime.now()

    timestamp = now.strftime("%Y-%m-%d %H:%M:%S")

    file_time = now.strftime("%Y%m%d_%H%M%S")

    base = script_dir()

    output = os.path.join(base, OUTPUT_FOLDER)

    os.makedirs(output, exist_ok=True)

    creds = load_env_file(os.path.join(base, CREDENTIALS_FILE))

    username = creds["GLOBAL_USERNAME"]

    password = creds["GLOBAL_PASSWORD"]

    auth = (username, password)

    with requests.Session() as session:

        for group in GROUPS:

            prefix = group["prefix"]

            cluster = group["cluster"]

            start = group["start"]

            end = group["end"]

            futures = []

            results: List[Result] = []

            with ThreadPoolExecutor(max_workers=MAX_WORKERS) as executor:

                for i in range(start, end + 1):

                    futures.append(

                        executor.submit(fetch, session, prefix, i, auth, timestamp, cluster)

                    )

                for f in as_completed(futures):

                    results.append(f.result())

            results.sort(key=lambda r: r.encolados, reverse=True)

            print("\nCLUSTER:", cluster)

            print_table(results)

            csv_name = f"encolados_{cluster}_{file_time}.csv"

            csv_path = os.path.join(output, csv_name)

            save_csv(results, csv_path)

            print("\nCSV guardado:", csv_path)


if __name__ == "__main__":

    main()
