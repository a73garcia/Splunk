#!/usr/bin/env python3
# -*- coding: utf-8 -*-

import os
import re
import time
import threading
from dataclasses import dataclass
from typing import Optional, List, Dict, Tuple
from datetime import datetime

import requests
import urllib3
import xml.etree.ElementTree as ET
from concurrent.futures import ThreadPoolExecutor, as_completed


# =========================
# Config
# =========================
CREDENTIALS_FILE = "credentials.env"

TIMEOUT_SECONDS = 12
MAX_WORKERS = 12
RETRIES = 2

REFRESH_SECONDS = 60

# Umbrales (por ENCOLADOS ACTUAL)
WARNING_THRESHOLD = 50
CRITICAL_THRESHOLD = 100

# SSL (ESA suele usar self-signed)
VERIFY_SSL = False
if not VERIFY_SSL:
    urllib3.disable_warnings(urllib3.exceptions.InsecureRequestWarning)

# ANSI colors
RED = "\033[91m"
YELLOW = "\033[93m"
RESET = "\033[0m"

# Clusters
GROUPS = [
    {"cluster": "Europa",       "pattern": "https://dh529-esa{idx}.c3s2.iphmx.com/xml/status", "start": 1, "end": 20},
    {"cluster": "Aplicaciones", "pattern": "https://dh5505-esa{idx}.c3s2.iphmx.com/xml/status", "start": 1, "end": 20},
    {"cluster": "América",      "pattern": "https://dh5504-esa{idx}.c3s2.iphmx.com/xml/status", "start": 1, "end": 20},
    {"cluster": "Brasil",       "pattern": "https://dh6121-esa{idx}.iphmx.com/xml/status",      "start": 1, "end": 20},
]


# =========================
# Model
# =========================
@dataclass
class Result:
    timestamp: str
    nodo: str
    cluster: str
    status: str

    encolados: int
    ram: int
    cpu: int
    conn_in: int
    conn_out: int
    active_recips: int

    error: Optional[str] = None


# =========================
# Helpers
# =========================
def script_dir() -> str:
    return os.path.dirname(os.path.abspath(__file__))


def load_env_file(path: str) -> Dict[str, str]:
    data: Dict[str, str] = {}
    if not os.path.exists(path):
        raise FileNotFoundError(f"No existe el archivo de credenciales: {path}")
    with open(path, "r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line or line.startswith("#") or "=" not in line:
                continue
            k, v = line.split("=", 1)
            data[k.strip()] = v.strip()
    return data


def nodo_from_index(idx: int) -> str:
    return f"ESA{idx:02d}"


def to_int(val: Optional[str]) -> int:
    if val is None:
        return 0
    s = str(val).strip()
    if not s or s.upper() == "N/A":
        return 0
    m = re.search(r"(\d+(\.\d+)?)", s)
    if not m:
        return 0
    try:
        return int(float(m.group(1)))
    except Exception:
        return 0


def fmt_num_dot(n: int) -> str:
    """
    Separador de miles con punto:
      1234567 -> 1.234.567
    """
    return f"{n:,}".replace(",", ".")


def trend_symbol(prev: int, cur: int) -> str:
    if cur > prev:
        return "+"
    if cur < prev:
        return "-"
    return "="


def colorize(text: str, encolados_actual: int, has_error: bool = False) -> str:
    if has_error:
        return text
    if encolados_actual >= CRITICAL_THRESHOLD:
        return f"{RED}{text}{RESET}"
    if encolados_actual >= WARNING_THRESHOLD:
        return f"{YELLOW}{text}{RESET}"
    return text


def clear_screen() -> None:
    # Limpieza ANSI
    print("\033[2J\033[H", end="")


# =========================
# XML parsing
# =========================
def parse_metrics(xml_text: str) -> Tuple[str, Dict[str, int]]:
    root = ET.fromstring(xml_text)

    system_status = "No disponible"
    sys_node = root.find(".//system")
    if sys_node is not None:
        system_status = sys_node.attrib.get("status", system_status)

    wanted = {
        "msgs_in_work_queue": 0,
        "ram_utilization": 0,
        "cpu_utilization": 0,
        "conn_in": 0,
        "conn_out": 0,
        "active_recips": 0,
    }

    for gauge in root.findall(".//gauge"):
        name = gauge.attrib.get("name")
        if name in wanted:
            val = gauge.attrib.get("current")
            if val is None:
                cur = gauge.find("./current")
                if cur is not None and cur.text:
                    val = cur.text.strip()
            wanted[name] = to_int(val)

    return system_status, wanted


# =========================
# Network
# =========================
def fetch_one(
    session: requests.Session,
    url: str,
    auth: Tuple[str, str],
    timestamp: str,
    cluster: str,
    idx: int,
) -> Result:
    node = nodo_from_index(idx)
    headers = {"Accept": "application/xml"}

    last_err: Optional[str] = None
    for _ in range(RETRIES + 1):
        try:
            r = session.get(
                url,
                headers=headers,
                auth=auth,
                timeout=TIMEOUT_SECONDS,
                verify=VERIFY_SSL,
            )

            if r.status_code != 200:
                return Result(timestamp, node, cluster, "Error conexión", 0, 0, 0, 0, 0, 0, f"HTTP {r.status_code}")

            status, m = parse_metrics(r.text)
            return Result(
                timestamp=timestamp,
                nodo=node,
                cluster=cluster,
                status=status,
                encolados=m["msgs_in_work_queue"],
                ram=m["ram_utilization"],
                cpu=m["cpu_utilization"],
                conn_in=m["conn_in"],
                conn_out=m["conn_out"],
                active_recips=m["active_recips"],
                error=None,
            )

        except requests.exceptions.RequestException as e:
            last_err = str(e)

    return Result(timestamp, node, cluster, "Error conexión", 0, 0, 0, 0, 0, 0, last_err or "RequestException")


# =========================
# UI: Table + Alerts
# =========================
def print_table(cluster: str, results: List[Result], prev_enc: Dict[str, int]) -> None:
    results_sorted = sorted(results, key=lambda r: r.encolados, reverse=True)

    headers = [
        "Timestamp", "Nodo", "Status",
        "Encolados_Ant", "Encolados_Act", "Var",
        "RAM", "CPU", "Conn_in", "Conn_out", "Active_recips",
        "Error"
    ]

    rows: List[List[str]] = []
    total_prev = 0
    total_cur = 0

    for r in results_sorted:
        prev = prev_enc.get(r.nodo, r.encolados)  # primera iteración prev=cur
        cur = r.encolados
        sym = trend_symbol(prev, cur)

        total_prev += prev
        total_cur += cur

        rows.append([
            r.timestamp,
            r.nodo,
            r.status,
            fmt_num_dot(prev),
            fmt_num_dot(cur),
            sym,
            fmt_num_dot(r.ram),
            fmt_num_dot(r.cpu),
            fmt_num_dot(r.conn_in),
            fmt_num_dot(r.conn_out),
            fmt_num_dot(r.active_recips),
            (r.error or "")
        ])

    total_sym = trend_symbol(total_prev, total_cur)
    total_row = ["", "TOTAL", "", fmt_num_dot(total_prev), fmt_num_dot(total_cur), total_sym, "", "", "", "", "", ""]

    widths = [len(h) for h in headers]
    for row in rows + [total_row]:
        for i, cell in enumerate(row):
            widths[i] = max(widths[i], len(str(cell)))

    def fmt(vals):
        return " | ".join(str(v).ljust(widths[i]) for i, v in enumerate(vals))

    sep = "-+-".join("-" * w for w in widths)

    print(f"=== {cluster} | refresco cada {REFRESH_SECONDS}s | Pulsa ESC para salir ===\n")
    print(fmt(headers))
    print(sep)

    for r, row in zip(results_sorted, rows):
        line = fmt(row)
        print(colorize(line, r.encolados, has_error=bool(r.error)))

    print(sep)
    print(colorize(fmt(total_row), total_cur, has_error=False))


def print_alerts(cluster: str, results: List[Result]) -> None:
    critical = [r for r in results if (not r.error and r.encolados >= CRITICAL_THRESHOLD)]
    warning = [r for r in results if (not r.error and WARNING_THRESHOLD <= r.encolados < CRITICAL_THRESHOLD)]

    if not critical and not warning:
        return

    print(f"\nALERTAS ({cluster}) - CRITICAL: {len(critical)} | WARNING: {len(warning)}")

    for r in sorted(critical, key=lambda x: x.encolados, reverse=True):
        print(f"{RED}CRITICAL{RESET} - {r.nodo} {cluster} -> encolados={fmt_num_dot(r.encolados)}")

    for r in sorted(warning, key=lambda x: x.encolados, reverse=True):
        print(f"{YELLOW}WARNING{RESET}  - {r.nodo} {cluster} -> encolados={fmt_num_dot(r.encolados)}")


# =========================
# ESC listener (cross-platform)
# =========================
def start_esc_listener(stop_event: threading.Event) -> None:
    try:
        import msvcrt  # type: ignore

        def _win_loop():
            while not stop_event.is_set():
                if msvcrt.kbhit():
                    ch = msvcrt.getch()
                    if ch == b"\x1b":
                        stop_event.set()
                        return
                time.sleep(0.05)

        threading.Thread(target=_win_loop, daemon=True).start()
        return

    except Exception:
        pass

    import sys
    import select
    import termios
    import tty

    def _posix_loop():
        fd = sys.stdin.fileno()
        old = termios.tcgetattr(fd)
        try:
            tty.setcbreak(fd)
            while not stop_event.is_set():
                r, _, _ = select.select([sys.stdin], [], [], 0.1)
                if r:
                    ch = sys.stdin.read(1)
                    if ch == "\x1b":
                        stop_event.set()
                        return
        finally:
            termios.tcsetattr(fd, termios.TCSADRAIN, old)

    threading.Thread(target=_posix_loop, daemon=True).start()


# =========================
# App
# =========================
def choose_cluster() -> Dict:
    names = [g["cluster"] for g in GROUPS]

    print("Clusters disponibles:")
    for i, n in enumerate(names, 1):
        print(f"  {i}) {n}")

    while True:
        choice = input("\nElige cluster (nombre o número): ").strip()

        if choice.isdigit():
            idx = int(choice)
            if 1 <= idx <= len(GROUPS):
                return GROUPS[idx - 1]

        for g in GROUPS:
            if g["cluster"].lower() == choice.lower():
                return g

        print("Opción no válida. Prueba de nuevo.")


def run_loop_for_cluster(group: Dict, auth: Tuple[str, str]) -> None:
    cluster = group["cluster"]
    pattern = group["pattern"]
    start = int(group["start"])
    end = int(group["end"])

    stop_event = threading.Event()
    start_esc_listener(stop_event)

    prev_encolados: Dict[str, int] = {}

    # Limpieza al entrar al cluster seleccionado
    clear_screen()

    with requests.Session() as session:
        while not stop_event.is_set():
            now = datetime.now()
            ts_col = now.strftime("%Y-%m-%d %H:%M:%S")

            targets = [(i, pattern.format(idx=i)) for i in range(start, end + 1)]

            results: List[Result] = []
            with ThreadPoolExecutor(max_workers=MAX_WORKERS) as ex:
                futures = [
                    ex.submit(fetch_one, session, url, auth, ts_col, cluster, idx)
                    for idx, url in targets
                ]
                for fut in as_completed(futures):
                    results.append(fut.result())

            # Limpieza en cada actualización
            clear_screen()
            print_table(cluster, results, prev_encolados)
            print_alerts(cluster, results)
            print("\n(Refrescando... pulsa ESC para salir)")

            prev_encolados = {r.nodo: r.encolados for r in results}

            # Sleep en trozos para reaccionar rápido al ESC
            for _ in range(REFRESH_SECONDS * 10):
                if stop_event.is_set():
                    break
                time.sleep(0.1)

    clear_screen()
    print(f"Salida solicitada. Monitor detenido para cluster: {cluster}")


def main() -> int:
    base = script_dir()
    creds = load_env_file(os.path.join(base, CREDENTIALS_FILE))

    username = creds.get("GLOBAL_USERNAME")
    password = creds.get("GLOBAL_PASSWORD")
    if not username or not password:
        print("Faltan GLOBAL_USERNAME y/o GLOBAL_PASSWORD en credentials.env")
        return 2

    group = choose_cluster()

    # Limpieza tras seleccionar opción del menú
    clear_screen()

    run_loop_for_cluster(group, (username, password))
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
