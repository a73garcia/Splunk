CREDENTIALS_FILE = "credentials.env"

TIMEOUT_SECONDS = 12
MAX_WORKERS = 12
RETRIES = 2
REFRESH_SECONDS = 60

VERIFY_SSL = False
if not VERIFY_SSL:
    urllib3.disable_warnings(urllib3.exceptions.InsecureRequestWarning)

RED = "\033[91m"
YELLOW = "\033[93m"
GREEN = "\033[92m"
RESET = "\033[0m"

GROUPS = [
    {"cluster": "Europa",       "pattern": "https://dh529-esa{idx}.c3s2.iphmx.com/xml/status", "start": 1, "end": 20},
    {"cluster": "América",      "pattern": "https://dh5504-esa{idx}.c3s2.iphmx.com/xml/status", "start": 1, "end": 20},
    {"cluster": "Brasil",       "pattern": "https://dh6121-esa{idx}.iphmx.com/xml/status",      "start": 1, "end": 20},
    {"cluster": "Aplicaciones", "pattern": "https://dh5505-esa{idx}.c3s2.iphmx.com/xml/status", "start": 1, "end": 20},
]

CLUSTER_THRESHOLDS = {
    "Europa": {
        "warning": 50,
        "critical": 100,
        "total_warning": 200,
        "total_critical": 300,
    },
    "América": {
        "warning": 50,
        "critical": 100,
        "total_warning": 200,
        "total_critical": 300,
    },
    "Brasil": {
        "warning": 50,
        "critical": 100,
        "total_warning": 200,
        "total_critical": 300,
    },
    "Aplicaciones": {
        "warning": 20,
        "critical": 40,
        "total_warning": 80,
        "total_critical": 120,
    },
}

stop_event = threading.Event()


@dataclass
class Result:
    timestamp: str
    nodo: str
    cluster: str
    status: str
    encolados: int
    error: Optional[str] = None


# =========================
# Helpers
# =========================
def clear_screen() -> None:
    os.system("cls")


def load_env_file(path: str) -> Dict[str, str]:
    data: Dict[str, str] = {}
    with open(path, "r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line or line.startswith("#") or "=" not in line:
                continue
            k, v = line.split("=", 1)
            data[k.strip()] = v.strip()
    return data


def nodo_from_index(idx: int) -> str:
    return f"ESA{idx:02d}"


def fmt_num_dot(n: int) -> str:
    return f"{n:,}".replace(",", ".")


def trend_symbol(prev: int, cur: int) -> str:
    if cur > prev:
        return "↑"
    if cur < prev:
        return "↓"
    return "→"


def colorize_node(text: str, cluster: str, current: int, has_error: bool = False) -> str:
    if has_error:
        return text

    th = CLUSTER_THRESHOLDS[cluster]

    if current >= th["critical"]:
        return f"{RED}{text}{RESET}"
    if current >= th["warning"]:
        return f"{YELLOW}{text}{RESET}"
    return text


def total_status_colored(cluster: str, total: int) -> str:
    th = CLUSTER_THRESHOLDS[cluster]

    if total >= th["total_critical"]:
        return f"{RED}{'CRITICAL':<8}{RESET}"
    if total >= th["total_warning"]:
        return f"{YELLOW}{'WARNING ':<8}{RESET}"
    return f"{GREEN}{'OK':^8}{RESET}"


def colorize_total(text: str, cluster: str, value: int) -> str:
    th = CLUSTER_THRESHOLDS[cluster]

    if value >= th["total_critical"]:
        return f"{RED}{text}{RESET}"
    if value >= th["total_warning"]:
        return f"{YELLOW}{text}{RESET}"
    return text


# =========================
# ESC listener (Windows)
# =========================
def esc_listener() -> None:
    while not stop_event.is_set():
        if msvcrt.kbhit():
            key = msvcrt.getch()
            if key == b"\x1b":
                stop_event.set()
                return
        time.sleep(0.1)


# =========================
# XML parser
# =========================
def parse_encolados(xml_text: str) -> Tuple[str, int]:
    root = ET.fromstring(xml_text)

    system_status = "No disponible"
    sys_node = root.find(".//system")
    if sys_node is not None:
        system_status = sys_node.attrib.get("status", system_status)

    encolados = 0
    for gauge in root.findall(".//gauge"):
        if gauge.attrib.get("name") == "msgs_in_work_queue":
            val = gauge.attrib.get("current")
            if val is None:
                cur = gauge.find("./current")
                if cur is not None and cur.text:
                    val = cur.text.strip()
            try:
                encolados = int(float(val))
            except Exception:
                encolados = 0
            break

    return system_status, encolados


# =========================
# Network
# =========================
def fetch_one(
    session: requests.Session,
    url: str,
    auth: Tuple[str, str],
    timestamp: str,
    cluster: str,
    idx: int,
) -> Result:
    nodo = nodo_from_index(idx)

    last_err: Optional[str] = None
    for _ in range(RETRIES + 1):
        try:
            r = session.get(
                url,
                auth=auth,
                timeout=TIMEOUT_SECONDS,
                verify=VERIFY_SSL,
                headers={"Accept": "application/xml"},
            )

            if r.status_code != 200:
                return Result(timestamp, nodo, cluster, "ERR", 0, f"HTTP {r.status_code}")

            status, encolados = parse_encolados(r.text)
            return Result(timestamp, nodo, cluster, status, encolados, None)

        except requests.exceptions.RequestException as e:
            last_err = str(e)

    return Result(timestamp, nodo, cluster, "ERR", 0, last_err or "RequestException")


def fetch_cluster(
    session: requests.Session,
    group: Dict,
    auth: Tuple[str, str],
    timestamp: str,
) -> List[Result]:
    cluster = group["cluster"]
    pattern = group["pattern"]
    start = int(group["start"])
    end = int(group["end"])

    results: List[Result] = []
    with ThreadPoolExecutor(max_workers=MAX_WORKERS) as ex:
        futures = [
            ex.submit(fetch_one, session, pattern.format(idx=i), auth, timestamp, cluster, i)
            for i in range(start, end + 1)
        ]
        for fut in as_completed(futures):
            results.append(fut.result())

    return sorted(results, key=lambda r: r.nodo)


# =========================
# Main combined table
# =========================
def print_combined_table(
    timestamp: str,
    current_data: Dict[str, Dict[str, Result]],
    previous_data: Dict[str, Dict[str, int]],
) -> None:
    cluster_order = ["Europa", "América", "Brasil", "Aplicaciones"]
    nodos = [f"ESA{i:02d}" for i in range(1, 21)]

    ts_width = 8
    nodo_width = 6

    ant_w = 9
    act_w = 9
    var_w = 5

    col_sep = " | "
    cluster_sep = " || "

    cluster_chunk_width = ant_w + len(col_sep) + act_w + len(col_sep) + var_w

    def fmt_chunk(prev_val: int, cur_val: int, var_val: str) -> str:
        return (
            f"{fmt_num_dot(prev_val).rjust(ant_w)}{col_sep}"
            f"{fmt_num_dot(cur_val).rjust(act_w)}{col_sep}"
            f"{var_val.center(var_w)}"
        )

    h1 = (
        f"| {'Hora'.ljust(ts_width)}{col_sep}"
        f"{'Nodo'.ljust(nodo_width)}{cluster_sep}"
        f"{'Europa'.center(cluster_chunk_width)}{cluster_sep}"
        f"{'América'.center(cluster_chunk_width)}{cluster_sep}"
        f"{'Brasil'.center(cluster_chunk_width)}{cluster_sep}"
        f"{'Aplicaciones'.center(cluster_chunk_width)} |"
    )

    sub = (
        f"{'Anterior'.center(ant_w)}{col_sep}"
        f"{'Encolados'.center(act_w)}{col_sep}"
        f"{'Var'.center(var_w)}"
    )

    h2 = (
        f"| {''.ljust(ts_width)}{col_sep}"
        f"{''.ljust(nodo_width)}{cluster_sep}"
        f"{sub}{cluster_sep}{sub}{cluster_sep}{sub}{cluster_sep}{sub} |"
    )

    sep = "-" * max(len(h1), len(h2))

    print(h1)
    print(h2)
    print(sep)

    for nodo in nodos:
        line = f"| {timestamp.ljust(ts_width)}{col_sep}{nodo.ljust(nodo_width)}"

        for cluster in cluster_order:
            cur_map = current_data.get(cluster, {})
            prev_map = previous_data.get(cluster, {})

            cur = cur_map.get(nodo)
            cur_val = cur.encolados if cur else 0
            prev_val = prev_map.get(nodo, cur_val)
            var = trend_symbol(prev_val, cur_val)
            has_error = bool(cur.error) if cur else False

            chunk = fmt_chunk(prev_val, cur_val, var)
            chunk = colorize_node(chunk, cluster, cur_val, has_error=has_error)
            line += f"{cluster_sep}{chunk}"

        line += " |"
        print(line)

    print(sep)

    total_line = f"| {''.ljust(ts_width)}{col_sep}{'TOTAL'.ljust(nodo_width)}"

    for cluster in cluster_order:
        cur_map = current_data.get(cluster, {})
        prev_map = previous_data.get(cluster, {})

        total_cur = sum(r.encolados for r in cur_map.values())
        total_prev = sum(
            prev_map.get(n, cur_map.get(n).encolados if cur_map.get(n) else 0)
            for n in nodos
        )
        var = trend_symbol(total_prev, total_cur)

        chunk = fmt_chunk(total_prev, total_cur, var)
        chunk = colorize_total(chunk, cluster, total_cur)
        total_line += f"{cluster_sep}{chunk}"

    total_line += " |"
    print(total_line)


# =========================
# Summary panel
# =========================
def print_cluster_summary_panel(current_data: Dict[str, Dict[str, Result]]) -> None:
    print("RESUMEN CLUSTERS\n")

    headers = ["Cluster", "Total", "Estado"]
    c1 = 13
    c2 = 10
    c3 = 8

    header = f"{headers[0].ljust(c1)} | {headers[1].rjust(c2)} | {headers[2].center(c3)}"
    sep = "-" * len(header)

    print(header)
    print(sep)

    for group in GROUPS:
        cluster = group["cluster"]
        total = sum(r.encolados for r in current_data.get(cluster, {}).values())
        status_colored = total_status_colored(cluster, total)

        print(f"{cluster.ljust(c1)} | {fmt_num_dot(total).rjust(c2)} | {status_colored}")


# =========================
# Top / alerts
# =========================
def print_top_nodes(current_data: Dict[str, Dict[str, Result]]) -> None:
    nodos = [f"ESA{i:02d}" for i in range(1, 21)]

    totals: Dict[str, int] = {}
    for nodo in nodos:
        total = 0
        for cluster_map in current_data.values():
            r = cluster_map.get(nodo)
            if r:
                total += r.encolados
        totals[nodo] = total

    top = sorted(totals.items(), key=lambda x: x[1], reverse=True)[:5]

    print("\nTOP 5 NODOS\n")
    for rank, (nodo, value) in enumerate(top, start=1):
        text = f"{rank}. {nodo} → {fmt_num_dot(value)}"
        if value >= 100:
            print(f"{RED}{text}{RESET}")
        elif value >= 50:
            print(f"{YELLOW}{text}{RESET}")
        else:
            print(text)


def print_alerts(current_data: Dict[str, Dict[str, Result]]) -> None:
    alerts = []

    for cluster, nodo_map in current_data.items():
        th = CLUSTER_THRESHOLDS[cluster]

        for nodo, r in nodo_map.items():
            if r.error:
                continue
            if r.encolados >= th["critical"]:
                alerts.append(("CRITICAL", cluster, nodo, r.encolados))
            elif r.encolados >= th["warning"]:
                alerts.append(("WARNING ", cluster, nodo, r.encolados))

    print("\nALERTAS NODOS\n")

    if not alerts:
        print(f"{GREEN}OK - Sin alertas{RESET}")
        return

    for level, cluster, nodo, value in sorted(alerts, key=lambda x: x[3], reverse=True):
        if level.strip() == "CRITICAL":
            print(f"{RED}{level}{RESET} {cluster} {nodo} → {fmt_num_dot(value)}")
        else:
            print(f"{YELLOW}{level}{RESET} {cluster} {nodo} → {fmt_num_dot(value)}")


def print_total_alerts(current_data: Dict[str, Dict[str, Result]]) -> None:
    print("\nALERTAS TOTALES CLUSTER\n")

    for group in GROUPS:
        cluster = group["cluster"]
        total = sum(r.encolados for r in current_data.get(cluster, {}).values())
        status = total_status_colored(cluster, total)
        print(f"{cluster:<13} → {status} | Total={fmt_num_dot(total)}")


# =========================
# Main loop
# =========================
def run_monitor(auth: Tuple[str, str]) -> None:
    previous_data: Dict[str, Dict[str, int]] = {}

    listener = threading.Thread(target=esc_listener, daemon=True)
    listener.start()

    with requests.Session() as session:
        while not stop_event.is_set():
            timestamp = datetime.now().strftime("%H:%M:%S")
            current_data: Dict[str, Dict[str, Result]] = {}

            with ThreadPoolExecutor(max_workers=len(GROUPS)) as ex:
                fut_map = {
                    ex.submit(fetch_cluster, session, g, auth, timestamp): g["cluster"]
                    for g in GROUPS
                }

                for fut in as_completed(fut_map):
                    cluster = fut_map[fut]
                    results = fut.result()
                    current_data[cluster] = {r.nodo: r for r in results}

            clear_screen()
            print("MONITOR COLAS ESA\n")
            print_cluster_summary_panel(current_data)
            print()
            print_combined_table(timestamp, current_data, previous_data)
            print_top_nodes(current_data)
            print_alerts(current_data)
            print_total_alerts(current_data)
            print("\nPulsa ESC para salir")

            previous_data = {
                c: {n: r.encolados for n, r in m.items()}
                for c, m in current_data.items()
            }

            for _ in range(REFRESH_SECONDS * 10):
                if stop_event.is_set():
                    break
                time.sleep(0.1)

    clear_screen()
    print("Monitor detenido (ESC pulsado)")


def main() -> None:
    creds = load_env_file(CREDENTIALS_FILE)
    auth = (creds["GLOBAL_USERNAME"], creds["GLOBAL_PASSWORD"])

    clear_screen()
    run_monitor(auth)


if __name__ == "__main__":
    main()

        print(line)

    print(sep)

    total_line = f"| {''.ljust(ts_width)}{col_sep}{'TOTAL'.ljust(nodo_width)}"

    for cluster in cluster_order:
        cur_map = current_data.get(cluster, {})
        prev_map = previous_data.get(cluster, {})

        total_cur = sum(r.encolados for r in cur_map.values())
        total_prev = sum(
            prev_map.get(n, cur_map.get(n).encolados if cur_map.get(n) else 0)
            for n in nodos
        )
        var = trend_symbol(total_prev, total_cur)

        chunk = fmt_chunk(total_prev, total_cur, var)
        chunk = colorize_total(chunk, cluster, total_cur)
        total_line += f"{cluster_sep}{chunk}"

    total_line += " |"
    print(total_line)

def print_top_nodes(current_data: Dict[str, Dict[str, Result]]) -> None:
    nodos = [f"ESA{i:02d}" for i in range(1, 21)]

    totals: Dict[str, int] = {}
    for nodo in nodos:
        total = 0
        for cluster_map in current_data.values():
            r = cluster_map.get(nodo)
            if r:
                total += r.encolados
        totals[nodo] = total

    top = sorted(totals.items(), key=lambda x: x[1], reverse=True)[:5]

    print("\nTOP 5 NODOS\n")
    for rank, (nodo, value) in enumerate(top, start=1):
        text = f"{rank}. {nodo} → {fmt_num_dot(value)}"
        if value >= CRITICAL_THRESHOLD:
            print(f"{RED}{text}{RESET}")
        elif value >= WARNING_THRESHOLD:
            print(f"{YELLOW}{text}{RESET}")
        else:
            print(text)


def print_alerts(current_data: Dict[str, Dict[str, Result]]) -> None:
    alerts = []

    for cluster, nodo_map in current_data.items():
        th = CLUSTER_THRESHOLDS[cluster]

        for nodo, r in nodo_map.items():
            if r.error:
                continue
            if r.encolados >= th["critical"]:
                alerts.append(("CRITICAL", cluster, nodo, r.encolados))
            elif r.encolados >= th["warning"]:
                alerts.append(("WARNING ", cluster, nodo, r.encolados))

    print("\nALERTAS NODOS\n")

    if not alerts:
        print(f"{GREEN}OK - Sin alertas{RESET}")
        return

    for level, cluster, nodo, value in sorted(alerts, key=lambda x: x[3], reverse=True):
        if level.strip() == "CRITICAL":
            print(f"{RED}{level}{RESET} {cluster} {nodo} → {fmt_num_dot(value)}")
        else:
            print(f"{YELLOW}{level}{RESET} {cluster} {nodo} → {fmt_num_dot(value)}")


def print_total_alerts(current_data: Dict[str, Dict[str, Result]]) -> None:
    print("\nALERTAS TOTALES CLUSTER\n")

    for group in GROUPS:
        cluster = group["cluster"]
        total = sum(r.encolados for r in current_data.get(cluster, {}).values())
        status = total_status_colored(cluster, total)
        print(f"{cluster:<13} → {status} | Total={fmt_num_dot(total)}")

def run_monitor(auth: Tuple[str, str]) -> None:
    previous_data: Dict[str, Dict[str, int]] = {}

    listener = threading.Thread(target=esc_listener, daemon=True)
    listener.start()

    with requests.Session() as session:
        while not stop_event.is_set():
            timestamp = datetime.now().strftime("%H:%M:%S")
            current_data: Dict[str, Dict[str, Result]] = {}

            with ThreadPoolExecutor(max_workers=len(GROUPS)) as ex:
                fut_map = {
                    ex.submit(fetch_cluster, session, g, auth, timestamp): g["cluster"]
                    for g in GROUPS
                }

                for fut in as_completed(fut_map):
                    cluster = fut_map[fut]
                    results = fut.result()
                    current_data[cluster] = {r.nodo: r for r in results}

            clear_screen()
            print("MONITOR COLAS ESA\n")
            print_cluster_summary_panel(current_data)
            print()
            print_combined_table(timestamp, current_data, previous_data)
            print_top_nodes(current_data)
            print_alerts(current_data)
            print_total_alerts(current_data)
            print("\nPulsa ESC para salir")

            previous_data = {
                c: {n: r.encolados for n, r in m.items()}
                for c, m in current_data.items()
            }

            for _ in range(REFRESH_SECONDS * 10):
                if stop_event.is_set():
                    break
                time.sleep(0.1)

    clear_screen()
    print("Monitor detenido (ESC pulsado)")


def main() -> None:
    creds = load_env_file(CREDENTIALS_FILE)
    auth = (creds["GLOBAL_USERNAME"], creds["GLOBAL_PASSWORD"])

    clear_screen()
    run_monitor(auth)


if __name__ == "__main__":
    main()
