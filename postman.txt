#!/usr/bin/env python3
# -*- coding: utf-8 -*-

import os
import csv
import re
from dataclasses import dataclass
from typing import Optional, List, Dict, Tuple
from datetime import datetime

import requests
import urllib3
import xml.etree.ElementTree as ET
from concurrent.futures import ThreadPoolExecutor, as_completed


# -------------------------
# Config
# -------------------------
CREDENTIALS_FILE = "credentials.env"
OUTPUT_FOLDER = "encolados"

TIMEOUT_SECONDS = 12
MAX_WORKERS = 12
RETRIES = 2

# Umbrales para coloreo/alerta (según ENCOLADOS)
WARNING_THRESHOLD = 50
CRITICAL_THRESHOLD = 100

# SSL (ESA suele usar self-signed)
VERIFY_SSL = False
if not VERIFY_SSL:
    urllib3.disable_warnings(urllib3.exceptions.InsecureRequestWarning)

# Colores consola (ANSI)
RED = "\033[91m"
YELLOW = "\033[93m"
RESET = "\033[0m"

# Clusters
GROUPS = [
    {"cluster": "Europa",       "pattern": "https://dh529-esa{idx}.c3s2.iphmx.com/xml/status", "start": 1, "end": 20},
    {"cluster": "Aplicaciones", "pattern": "https://dh5505-esa{idx}.c3s2.iphmx.com/xml/status", "start": 1, "end": 20},
    {"cluster": "América",      "pattern": "https://dh5504-esa{idx}.c3s2.iphmx.com/xml/status", "start": 1, "end": 20},
    {"cluster": "Brasil",       "pattern": "https://dh6121-esa{idx}.iphmx.com/xml/status",      "start": 1, "end": 20},
]


# -------------------------
# Data model
# -------------------------
@dataclass
class Result:
    timestamp: str
    nodo: str
    cluster: str
    status: str

    encolados: int

    ram: int
    cpu: int
    conn_in: int
    conn_out: int
    active_recips: int

    error: Optional[str] = None


# -------------------------
# Helpers
# -------------------------
def script_dir() -> str:
    return os.path.dirname(os.path.abspath(__file__))


def load_env_file(path: str) -> Dict[str, str]:
    data: Dict[str, str] = {}
    if not os.path.exists(path):
        raise FileNotFoundError(f"No existe el archivo de credenciales: {path}")

    with open(path, "r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line or line.startswith("#") or "=" not in line:
                continue
            k, v = line.split("=", 1)
            data[k.strip()] = v.strip()
    return data


def nodo_from_index(idx: int) -> str:
    # Solo para mostrar: ESA01..ESA20 (con 0 delante)
    return f"ESA{idx:02d}"


def safe_filename(name: str) -> str:
    return (
        name.replace(" ", "_")
        .replace("/", "_")
        .replace("\\", "_")
        .replace(":", "_")
        .replace("á", "a").replace("é", "e").replace("í", "i").replace("ó", "o").replace("ú", "u")
        .replace("Á", "A").replace("É", "E").replace("Í", "I").replace("Ó", "O").replace("Ú", "U")
    )


def monthly_csv_path(out_dir: str, cluster: str, now: datetime) -> str:
    month = now.strftime("%Y-%m")
    return os.path.join(out_dir, f"encolados_{safe_filename(cluster)}_{month}.csv")


def colorize(text: str, encolados: int, has_error: bool = False) -> str:
    if has_error:
        return text
    if encolados >= CRITICAL_THRESHOLD:
        return f"{RED}{text}{RESET}"
    if encolados >= WARNING_THRESHOLD:
        return f"{YELLOW}{text}{RESET}"
    return text


def to_int(val: Optional[str]) -> int:
    """
    Convierte a int valores típicos:
      "23" -> 23
      "309G" -> 309
      "N/A" o None -> 0
    """
    if val is None:
        return 0
    s = str(val).strip()
    if not s or s.upper() == "N/A":
        return 0
    # extrae primer número (entero o decimal)
    m = re.search(r"(\d+(\.\d+)?)", s)
    if not m:
        return 0
    try:
        return int(float(m.group(1)))
    except Exception:
        return 0


# -------------------------
# XML parsing
# -------------------------
def parse_metrics(xml_text: str) -> Tuple[str, Dict[str, int]]:
    """
    Extrae status del sistema y gauges:
      - msgs_in_work_queue
      - ram_utilization
      - cpu_utilization
      - conn_in
      - conn_out
      - active_recips

    En ESA vienen como:
      <gauge name="xxx" current="123"/>
    """
    root = ET.fromstring(xml_text)

    system_status = "No disponible"
    sys_node = root.find(".//system")
    if sys_node is not None:
        system_status = sys_node.attrib.get("status", system_status)

    wanted = {
        "msgs_in_work_queue": 0,
        "ram_utilization": 0,
        "cpu_utilization": 0,
        "conn_in": 0,
        "conn_out": 0,
        "active_recips": 0,
    }

    for gauge in root.findall(".//gauge"):
        name = gauge.attrib.get("name")
        if name in wanted:
            val = gauge.attrib.get("current")
            # fallback por si viniera como nodo <current>
            if val is None:
                cur = gauge.find("./current")
                if cur is not None and cur.text:
                    val = cur.text.strip()
            wanted[name] = to_int(val)

    return system_status, wanted


# -------------------------
# Network
# -------------------------
def fetch_one(
    session: requests.Session,
    url: str,
    auth: Tuple[str, str],
    timestamp: str,
    cluster: str,
    idx: int,
) -> Result:
    node = nodo_from_index(idx)
    headers = {"Accept": "application/xml"}

    last_err: Optional[str] = None
    for _ in range(RETRIES + 1):
        try:
            r = session.get(
                url,
                headers=headers,
                auth=auth,
                timeout=TIMEOUT_SECONDS,
                verify=VERIFY_SSL,
            )

            if r.status_code != 200:
                return Result(
                    timestamp=timestamp,
                    nodo=node,
                    cluster=cluster,
                    status="Error conexión",
                    encolados=0,
                    ram=0, cpu=0, conn_in=0, conn_out=0, active_recips=0,
                    error=f"HTTP {r.status_code}",
                )

            status, m = parse_metrics(r.text)
            return Result(
                timestamp=timestamp,
                nodo=node,
                cluster=cluster,
                status=status,
                encolados=m["msgs_in_work_queue"],
                ram=m["ram_utilization"],
                cpu=m["cpu_utilization"],
                conn_in=m["conn_in"],
                conn_out=m["conn_out"],
                active_recips=m["active_recips"],
                error=None,
            )

        except requests.exceptions.RequestException as e:
            last_err = str(e)

    return Result(
        timestamp=timestamp,
        nodo=node,
        cluster=cluster,
        status="Error conexión",
        encolados=0,
        ram=0, cpu=0, conn_in=0, conn_out=0, active_recips=0,
        error=last_err or "RequestException",
    )


# -------------------------
# Output: CSV (append mensual)
# -------------------------
def append_csv(results: List[Result], path: str) -> None:
    fieldnames = [
        "timestamp", "nodo", "cluster", "status",
        "encolados", "ram", "cpu", "conn_in", "conn_out", "active_recips",
        "error"
    ]
    file_exists = os.path.exists(path)

    with open(path, "a", newline="", encoding="utf-8") as f:
        w = csv.DictWriter(f, fieldnames=fieldnames)
        if not file_exists:
            w.writeheader()

        for r in results:
            w.writerow({
                "timestamp": r.timestamp,
                "nodo": r.nodo,
                "cluster": r.cluster,
                "status": r.status,
                "encolados": r.encolados,
                "ram": r.ram,
                "cpu": r.cpu,
                "conn_in": r.conn_in,
                "conn_out": r.conn_out,
                "active_recips": r.active_recips,
                "error": r.error or "",
            })


def print_cluster_alerts(cluster: str, results: List[Result]) -> None:
    critical = [r for r in results if (not r.error and r.encolados >= CRITICAL_THRESHOLD)]
    warning = [r for r in results if (not r.error and WARNING_THRESHOLD <= r.encolados < CRITICAL_THRESHOLD)]

    if not critical and not warning:
        return

    print(f"\nALERTAS ({cluster})")
    for r in sorted(critical, key=lambda x: x.encolados, reverse=True):
        print(f"{RED}CRITICAL{RESET} - {r.nodo} {cluster} -> encolados={r.encolados} ({r.timestamp})")
    for r in sorted(warning, key=lambda x: x.encolados, reverse=True):
        print(f"{YELLOW}WARNING{RESET}  - {r.nodo} {cluster} -> encolados={r.encolados} ({r.timestamp})")


# -------------------------
# TABLAS POR CLUSTER (misma tabla, pero ampliada con métricas)
# -------------------------
def print_table(cluster: str, results: List[Result]) -> None:
    """
    Tabla detallada del cluster. Se colorea la fila completa según ENCOLADOS.
    """
    results_sorted = sorted(results, key=lambda r: r.encolados, reverse=True)

    headers = [
        "Timestamp", "Nodo", "Cluster", "Status",
        "Encolados", "RAM", "CPU", "Conn_in", "Conn_out", "Active_recips",
        "Error"
    ]

    rows = [
        [
            r.timestamp,
            r.nodo,
            r.cluster,
            r.status,
            str(r.encolados),
            str(r.ram),
            str(r.cpu),
            str(r.conn_in),
            str(r.conn_out),
            str(r.active_recips),
            (r.error or "")
        ]
        for r in results_sorted
    ]

    widths = [len(h) for h in headers]
    for row in rows:
        for i, cell in enumerate(row):
            widths[i] = max(widths[i], len(str(cell)))

    def fmt(vals):
        return " | ".join(str(v).ljust(widths[i]) for i, v in enumerate(vals))

    sep = "-+-".join("-" * w for w in widths)

    print(f"\n=== {cluster} ===")
    print(fmt(headers))
    print(sep)

    for r in results_sorted:
        line = fmt([
            r.timestamp, r.nodo, r.cluster, r.status,
            str(r.encolados), str(r.ram), str(r.cpu), str(r.conn_in), str(r.conn_out), str(r.active_recips),
            (r.error or "")
        ])
        print(colorize(line, r.encolados, has_error=bool(r.error)))


# -------------------------
# RESUMEN CLUSTERS (solo basado en ENCOLADOS)
# - sin Rank
# - Top3 en 3 filas
# - separador por cluster
# - colores según ENCOLADOS
# -------------------------
def print_summary(all_results: Dict[str, List[Result]]) -> None:
    print("\n============== RESUMEN CLUSTERS ==============")

    headers = ["Cluster", "Total Encolados", "Nodo", "Encolados", "Errores"]

    rows_with_separators: List[Tuple[str, Optional[List[str]]]] = []

    for cluster, results in all_results.items():
        total = sum(r.encolados for r in results)
        errors = sum(1 for r in results if r.error)
        sorted_nodes = sorted(results, key=lambda r: r.encolados, reverse=True)

        for pos in range(3):
            if pos < len(sorted_nodes):
                r = sorted_nodes[pos]
                row = [
                    cluster if pos == 0 else "",
                    str(total) if pos == 0 else "",
                    r.nodo,
                    str(r.encolados),
                    str(errors) if pos == 0 else ""
                ]
            else:
                row = [
                    cluster if pos == 0 else "",
                    str(total) if pos == 0 else "",
                    "-",
                    "0",
                    str(errors) if pos == 0 else ""
                ]
            rows_with_separators.append(("ROW", row))

        rows_with_separators.append(("SEP", None))

    rows_only = [row for kind, row in rows_with_separators if kind == "ROW" and row is not None]

    widths = [len(h) for h in headers]
    for row in rows_only:
        for i, cell in enumerate(row):
            widths[i] = max(widths[i], len(cell))

    def fmt(vals: List[str]) -> str:
        return " | ".join(vals[i].ljust(widths[i]) for i in range(len(vals)))

    sep_line = "-+-".join("-" * w for w in widths)
    cluster_sep_line = "=" * len(sep_line)

    print(fmt(headers))
    print(sep_line)

    for kind, row in rows_with_separators:
        if kind == "SEP":
            print(cluster_sep_line)
            continue

        assert row is not None
        encolados_val = to_int(row[3])
        line = fmt(row)
        print(colorize(line, encolados_val, has_error=False))


# -------------------------
# Main
# -------------------------
def main() -> int:
    now = datetime.now()
    ts_col = now.strftime("%Y-%m-%d %H:%M:%S")

    base = script_dir()
    out_dir = os.path.join(base, OUTPUT_FOLDER)
    os.makedirs(out_dir, exist_ok=True)

    creds = load_env_file(os.path.join(base, CREDENTIALS_FILE))
    username = creds.get("GLOBAL_USERNAME")
    password = creds.get("GLOBAL_PASSWORD")
    if not username or not password:
        print("Faltan GLOBAL_USERNAME y/o GLOBAL_PASSWORD en credentials.env")
        return 2

    auth = (username, password)

    all_results: Dict[str, List[Result]] = {}

    with requests.Session() as session:
        for g in GROUPS:
            cluster = g["cluster"]
            pattern = g["pattern"]
            start = int(g["start"])
            end = int(g["end"])

            targets = [(i, pattern.format(idx=i)) for i in range(start, end + 1)]

            results: List[Result] = []
            with ThreadPoolExecutor(max_workers=MAX_WORKERS) as ex:
                futures = [
                    ex.submit(fetch_one, session, url, auth, ts_col, cluster, idx)
                    for idx, url in targets
                ]
                for fut in as_completed(futures):
                    results.append(fut.result())

            all_results[cluster] = results

            # Tablas detalladas por cluster (ahora con RAM/CPU/CONN/ACTIVE)
            print_table(cluster, results)
            print_cluster_alerts(cluster, results)

            # CSV mensual por cluster (append)
            csv_path = monthly_csv_path(out_dir, cluster, now)
            append_csv(results, csv_path)
            print(f"\n[{cluster}] Filas añadidas: {len(results)} | CSV mensual: {csv_path}")

    # Resumen final (Top3 por cluster)
    print_summary(all_results)

    return 0


if __name__ == "__main__":
    raise SystemExit(main())
