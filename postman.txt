import platform
import threading
import time

stop_event = threading.Event()

def esc_listener():

    system = platform.system()

    # WINDOWS
    if system == "Windows":
        import msvcrt

        while not stop_event.is_set():

            if msvcrt.kbhit():
                key = msvcrt.getch()

                if key == b'\x1b':   # ESC
                    stop_event.set()
                    return

            time.sleep(0.1)

    # LINUX / MAC
    else:

        import sys
        import select
        import termios
        import tty

        fd = sys.stdin.fileno()
        old_settings = termios.tcgetattr(fd)

        try:

            tty.setcbreak(fd)

            while not stop_event.is_set():

                r, _, _ = select.select([sys.stdin], [], [], 0.1)

                if r:
                    key = sys.stdin.read(1)

                    if key == '\x1b':
                        stop_event.set()
                        return

        finally:
            termios.tcsetattr(fd, termios.TCSADRAIN, old_settings)

def esc_listener():

    fd = sys.stdin.fileno()
    old_settings = termios.tcgetattr(fd)

    try:

        tty.setcbreak(fd)

        while not stop_event.is_set():

            r, _, _ = select.select([sys.stdin], [], [], 0.1)

            if r:
                ch = sys.stdin.read(1)
                if ch == "\x1b":  # ESC
                    stop_event.set()
                    return

    finally:
        termios.tcsetattr(fd, termios.TCSADRAIN, old_settings)

def load_env_file(path):

    data = {}

    with open(path) as f:
        for line in f:
            if "=" in line:
                k, v = line.strip().split("=", 1)
                data[k] = v

    return data


def nodo_from_index(idx):
    return f"ESA{idx:02d}"


def fmt_num_dot(n):
    return f"{n:,}".replace(",", ".")


def trend_symbol(prev, cur):

    if cur > prev:
        return "↑"
    if cur < prev:
        return "↓"
    return "→"

def parse_encolados(xml_text):

    for gauge in root.findall(".//gauge"):

        if gauge.attrib.get("name") == "msgs_in_work_queue":

            val = gauge.attrib.get("current")

            if val is None:
                cur = gauge.find("./current")
                if cur is not None:
                    val = cur.text

            return int(float(val))

    return 0

def fetch_one(session, url, auth, timestamp, cluster, idx):

    nodo = nodo_from_index(idx)

    try:

        r = session.get(url, auth=auth, timeout=TIMEOUT_SECONDS, verify=False)

        if r.status_code != 200:
            return Result(timestamp, nodo, cluster, "ERR", 0)

        encolados = parse_encolados(r.text)

        return Result(timestamp, nodo, cluster, "OK", encolados)

    except Exception:

        return Result(timestamp, nodo, cluster, "ERR", 0)


def fetch_cluster(session, group, auth, timestamp):

    cluster = group["cluster"]
    pattern = group["pattern"]

    start = group["start"]
    end = group["end"]

    results = []

    with ThreadPoolExecutor(max_workers=MAX_WORKERS) as ex:

        futures = [
            ex.submit(fetch_one, session, pattern.format(idx=i), auth, timestamp, cluster, i)
            for i in range(start, end + 1)
        ]

        for fut in as_completed(futures):
            results.append(fut.result())

    return sorted(results, key=lambda r: r.nodo)


# =========================
# TOP NODES
# =========================

def print_top_nodes(current_data):

    nodos = [f"ESA{i:02d}" for i in range(1, 21)]

    totals = {}

    for nodo in nodos:

        total = 0

        for cluster_map in current_data.values():

            r = cluster_map.get(nodo)

            if r:
                total += r.encolados

        totals[nodo] = total

    top = sorted(totals.items(), key=lambda x: x[1], reverse=True)[:5]

    print("\nTOP 5 NODOS CON MÁS ENCOLADOS\n")

    for rank, (nodo, value) in enumerate(top, start=1):

        text = f"{rank}. {nodo} → {fmt_num_dot(value)}"

        if value >= CRITICAL_THRESHOLD:
            print(f"{RED}{text}{RESET}")
        elif value >= WARNING_THRESHOLD:
            print(f"{YELLOW}{text}{RESET}")
        else:
            print(text)


# =========================
# ALERTS
# =========================

def print_alerts(current_data):

    alerts = []

    for cluster, nodo_map in current_data.items():

        for nodo, r in nodo_map.items():

            if r.encolados >= CRITICAL_THRESHOLD:
                alerts.append(("CRITICAL", cluster, nodo, r.encolados))

            elif r.encolados >= WARNING_THRESHOLD:
                alerts.append(("WARNING", cluster, nodo, r.encolados))

    print("\nALERTAS NODOS")

    if not alerts:
        print(f"{GREEN}OK - Sin alertas{RESET}")
        return

    for level, cluster, nodo, value in sorted(alerts, key=lambda x: x[3], reverse=True):

        if level == "CRITICAL":
            print(f"{RED}CRITICAL{RESET} {cluster} {nodo} → {fmt_num_dot(value)}")

        else:
            print(f"{YELLOW}WARNING{RESET} {cluster} {nodo} → {fmt_num_dot(value)}")


# =========================
# TOTAL ALERTS
# =========================

def print_total_alerts(current_data):

    print("\nALERTAS TOTALES CLUSTER\n")

    for group in GROUPS:

        cluster = group["cluster"]

        total = sum(r.encolados for r in current_data.get(cluster, {}).values())

        if total >= TOTAL_CRITICAL_THRESHOLD:
            status = f"{RED}CRITICAL{RESET}"

        elif total >= TOTAL_WARNING_THRESHOLD:
            status = f"{YELLOW}WARNING{RESET}"

        else:
            status = f"{GREEN}OK{RESET}"

        print(f"{cluster:<13} → {status} | Total={fmt_num_dot(total)}")
