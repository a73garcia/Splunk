#!/usr/bin/env python3
# -*- coding: utf-8 -*-

import os
import time
import threading
import msvcrt
from dataclasses import dataclass
from typing import Optional, Dict
from datetime import datetime
import xml.etree.ElementTree as ET
from concurrent.futures import ThreadPoolExecutor, as_completed

import requests
import urllib3

urllib3.disable_warnings()

CREDENTIALS_FILE = "credentials.env"

TIMEOUT = 12
REFRESH = 60
MAX_WORKERS = 12

RED = "\033[91m"
YELLOW = "\033[93m"
GREEN = "\033[92m"
ORANGE = "\033[38;5;208m"
RESET = "\033[0m"

GROUPS = [
    {"cluster": "Europa", "pattern": "https://dh529-esa{idx}.c3s2.iphmx.com/xml/status"},
    {"cluster": "América", "pattern": "https://dh5504-esa{idx}.c3s2.iphmx.com/xml/status"},
    {"cluster": "Brasil", "pattern": "https://dh6121-esa{idx}.iphmx.com/xml/status"},
    {"cluster": "Aplicaciones", "pattern": "https://dh5505-esa{idx}.c3s2.iphmx.com/xml/status"},
]

stop_event = threading.Event()


@dataclass
class Result:
    nodo: str
    cluster: str
    status: str
    encolados: int
    cpu: int
    ram: int
    conn_in: int
    conn_out: int
    active_recips: int
    error: Optional[str] = None


# ---------------------------------------------------------
# helpers
# ---------------------------------------------------------

def clear():
    os.system("cls")


def separator():
    print(f"{ORANGE}{'═'*80}{RESET}")


def fmt(n):
    return f"{n:,}".replace(",", ".")


def load_credentials():

    creds = {}

    with open(CREDENTIALS_FILE) as f:

        for line in f:

            if "=" in line:

                k,v = line.strip().split("=",1)

                creds[k] = v

    return creds


# ---------------------------------------------------------
# XML parsing
# ---------------------------------------------------------

def parse_xml(xml):

    root = ET.fromstring(xml)

    status = "unknown"

    sys = root.find(".//system")

    if sys is not None:

        status = sys.attrib.get("status",status)

    gauges = {
        "msgs_in_work_queue":0,
        "cpu_utilization":0,
        "ram_utilization":0,
        "conn_in":0,
        "conn_out":0,
        "active_recips":0,
    }

    for g in root.findall(".//gauge"):

        name = g.attrib.get("name")

        if name in gauges:

            try:
                gauges[name] = int(float(g.attrib.get("current")))
            except:
                gauges[name] = 0

    return status,gauges


# ---------------------------------------------------------
# network
# ---------------------------------------------------------

def fetch_node(session,url,auth,cluster,idx):

    nodo=f"ESA{idx:02d}"

    try:

        r=session.get(url,auth=auth,timeout=TIMEOUT,verify=False)

        if r.status_code!=200:

            return Result(nodo,cluster,"offline",0,0,0,0,0,0,"HTTP")

        status,g=parse_xml(r.text)

        return Result(
            nodo,
            cluster,
            status,
            g["msgs_in_work_queue"],
            g["cpu_utilization"],
            g["ram_utilization"],
            g["conn_in"],
            g["conn_out"],
            g["active_recips"]
        )

    except:

        return Result(nodo,cluster,"offline",0,0,0,0,0,0,"ERROR")


def fetch_cluster(session,group,auth):

    cluster=group["cluster"]
    pattern=group["pattern"]

    results=[]

    with ThreadPoolExecutor(max_workers=MAX_WORKERS) as ex:

        futures=[]

        for i in range(1,21):

            url=pattern.format(idx=i)

            futures.append(ex.submit(fetch_node,session,url,auth,cluster,i))

        for f in as_completed(futures):

            results.append(f.result())

    return results


# ---------------------------------------------------------
# visual panels
# ---------------------------------------------------------

def print_cluster_summary(data,prev):

    print("RESUMEN CLUSTERS")
    print("─"*80)

    print(f"{'Cluster':<15}{'Total':>10}{'Estado':>12}{'Variación':>12}")

    for cluster,nodes in data.items():

        total=sum(r.encolados for r in nodes.values())

        prev_total=sum(prev.get(cluster,{}).values())

        diff=total-prev_total

        var=f"+{fmt(diff)}" if diff>0 else fmt(diff)

        state="OK"

        if total>200:

            state=f"{YELLOW}WARNING{RESET}"

        if total>300:

            state=f"{RED}CRITICAL{RESET}"

        print(f"{cluster:<15}{fmt(total):>10}{state:>12}{var:>12}")

    print()


def print_alerts(data):

    print("ALERTAS NODOS")
    print("─"*80)

    found=False

    for cluster,nodes in data.items():

        for nodo,r in nodes.items():

            if r.encolados>100:

                print(f"{RED}{cluster} {nodo} cola={fmt(r.encolados)}{RESET}")

                found=True

    if not found:

        print(f"{GREEN}Sin alertas{RESET}")

    print()


def print_risks(data,prev):

    print("NODOS EN RIESGO")
    print("─"*80)

    found=False

    for cluster,nodes in data.items():

        for nodo,r in nodes.items():

            prev_val=prev.get(cluster,{}).get(nodo,r.encolados)

            if r.encolados-prev_val>20:

                print(f"{YELLOW}{cluster} {nodo} creciendo{RESET}")

                found=True

    if not found:

        print(f"{GREEN}Sin nodos en riesgo{RESET}")

    print()

# tables

def print_queue_table(ts,data,prev):

    print("ESTADO ENCOLADOS NODOS\n")

    for i in range(1,21):

        nodo=f"ESA{i:02d}"

        row=f"{ts} {nodo}"

        for cluster in ["Europa","América","Brasil","Aplicaciones"]:

            r=data.get(cluster,{}).get(nodo)

            if r:

                prev_val=prev.get(cluster,{}).get(nodo,r.encolados)

                diff=r.encolados-prev_val

                sym="↑" if diff>0 else "↓" if diff<0 else "→"

                row+=f"  {fmt(prev_val)} {fmt(r.encolados)} {sym} {fmt(r.active_recips)}"

        print(row)


def print_health_table(data):

    print("\nSALUD OPERATIVA\n")

    for i in range(1,21):

        nodo=f"ESA{i:02d}"

        row=nodo

        for cluster in ["Europa","América","Brasil","Aplicaciones"]:

            r=data.get(cluster,{}).get(nodo)

            if r:

                status="online"

                if r.status.lower()!="online" or r.error:

                    status="offline"

                row+=f"  {status} {r.cpu} {r.ram} {r.conn_out}"

        print(row)


# ---------------------------------------------------------
# incidents
# ---------------------------------------------------------

def print_incidents(data):

    print("\nINCIDENCIAS\n")

    found=False

    for cluster,nodes in data.items():

        for nodo,r in nodes.items():

            if r.error:

                print(f"{RED}{cluster} {nodo} sin respuesta{RESET}")

                found=True

            elif r.status.lower()!="online":

                print(f"{YELLOW}{cluster} {nodo} status={r.status}{RESET}")

                found=True

    if not found:

        print(f"{GREEN}Sin incidencias{RESET}")


# ---------------------------------------------------------
# ESC listener
# ---------------------------------------------------------

def esc_listener():

    while not stop_event.is_set():

        if msvcrt.kbhit():

            if msvcrt.getch()==b"\x1b":

                stop_event.set()

        time.sleep(0.1)


# ---------------------------------------------------------
# monitor
# ---------------------------------------------------------

def run_monitor(auth):

    prev={}

    listener=threading.Thread(target=esc_listener,daemon=True)

    listener.start()

    with requests.Session() as session:

        while not stop_event.is_set():

            ts=datetime.now().strftime("%H:%M")

            data={}

            with ThreadPoolExecutor(max_workers=4) as ex:

                futures={ex.submit(fetch_cluster,session,g,auth):g["cluster"] for g in GROUPS}

                for f in as_completed(futures):

                    cluster=futures[f]

                    results=f.result()

                    data[cluster]={r.nodo:r for r in results}

            clear()

            print("MONITOR ESA\n")

            print_cluster_summary(data,prev)

            print_alerts(data)

            print_risks(data,prev)

            separator()

            print_queue_table(ts,data,prev)

            separator()

            print_health_table(data)

            print_incidents(data)

            prev={c:{n:r.encolados for n,r in m.items()} for c,m in data.items()}

            for _ in range(REFRESH*10):

                if stop_event.is_set():

                    break

                time.sleep(0.1)

    clear()

    print("Monitor detenido")


# ---------------------------------------------------------
# main
# ---------------------------------------------------------

def main():

    creds=load_credentials()

    auth=(creds["GLOBAL_USERNAME"],creds["GLOBAL_PASSWORD"])

    clear()

    run_monitor(auth)


if __name__=="__main__":

    main()
