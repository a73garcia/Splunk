def strip_ansi(text: str) -> str:
    s = str(text)
    for code in ANSI_CODES:
        s = s.replace(code, "")
    return s


def visible_text_len(text: str) -> int:
    return len(strip_ansi(str(text)))


def ansi_rjust(text: str, width: int) -> str:
    plain = strip_ansi(text)
    if len(plain) >= width:
        return text
    return (" " * (width - len(plain))) + text


def ansi_ljust(text: str, width: int) -> str:
    plain = strip_ansi(text)
    if len(plain) >= width:
        return text
    return text + (" " * (width - len(plain)))


def terminal_width(default: int = 180) -> int:
    try:
        return os.get_terminal_size().columns
    except OSError:
        return default


def pad_or_trim(text: str, width: int) -> str:
    text = str(text)
    plain = strip_ansi(text)
    if len(plain) <= width:
        return text + (" " * (width - len(plain)))
    if width <= 3:
        return plain[:width]
    return plain[: width - 3] + "..."


def block_max_width(lines: List[str], min_width: int = 20) -> int:
    if not lines:
        return min_width
    return max(min_width, max(visible_text_len(line) for line in lines))


def choose_three_col_widths(
    lines1: List[str],
    lines2: List[str],
    lines3: List[str],
    min1: int = 42,
    min2: int = 34,
    min3: int = 34,
    gap: int = 3,
) -> Tuple[int, int, int]:
    term_w = terminal_width()
    gaps_total = gap * 2
    available = max(80, term_w - gaps_total)

    need1 = block_max_width(lines1, min1)
    need2 = block_max_width(lines2, min2)
    need3 = block_max_width(lines3, min3)

    total_need = need1 + need2 + need3
    if total_need <= available:
        return need1, need2, need3

    mins = [min1, min2, min3]
    weights = [need1, need2, need3]
    extra_available = available - sum(mins)
    if extra_available <= 0:
        return mins[0], mins[1], mins[2]

    extra_needs = [max(0, w - m) for w, m in zip(weights, mins)]
    total_extra_need = sum(extra_needs)
    if total_extra_need == 0:
        return mins[0], mins[1], mins[2]

    assigned = []
    for extra_need, min_w in zip(extra_needs, mins):
        portion = int(extra_available * (extra_need / total_extra_need))
        assigned.append(min_w + portion)

    diff = available - sum(assigned)
    i = 0
    while diff > 0:
        assigned[i % 3] += 1
        diff -= 1
        i += 1

    return assigned[0], assigned[1], assigned[2]


def load_env_file(path: str) -> Dict[str, str]:
    data: Dict[str, str] = {}
    with open(path, "r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line or line.startswith("#") or "=" not in line:
                continue
            k, v = line.split("=", 1)
            data[k.strip()] = v.strip()
    return data


def nodo_from_index(idx: int) -> str:

def fmt_num_dot(n: int) -> str:

def to_int(value: Optional[str]) -> int:
    if value is None:
        return 0
    s = str(value).strip()
    if not s or s.upper() == "N/A":
        return 0

    digits = []
    for ch in s:
        if ch.isdigit() or ch == ".":
            digits.append(ch)
        else:
            break

    try:
        return int(float("".join(digits))) if digits else 0
    except Exception:
        return 0


def trend_symbol(prev: int, cur: int) -> str:
    if cur > prev:
        return "↑"
    if cur < prev:
        return "↓"
    return "="


def total_status_text(cluster: str, total: int) -> str:
    th = CLUSTER_THRESHOLDS[cluster]
    if total >= th["total_critical"]:
        return "CRITICAL"
    if total >= th["total_warning"]:
        return "WARNING"
    return "OK"


def total_status_colored(cluster: str, total: int) -> str:
    status = total_status_text(cluster, total)
    if status == "CRITICAL":
        return f"{RED}{status}{RESET}"
    if status == "WARNING":
        return f"{YELLOW}{status}{RESET}"
    return f"{GREEN}{status}{RESET}"

def health_score(r: Result, prev_queue: int) -> int:
    score = 100
    q_th = CLUSTER_THRESHOLDS[r.cluster]

    if r.encolados >= q_th["critical"]:
        score -= 35
    elif r.encolados >= q_th["warning"]:
        score -= 20

    if r.cpu >= ADV_THRESHOLDS["cpu_critical"]:
        score -= 20
    elif r.cpu >= ADV_THRESHOLDS["cpu_warning"]:
        score -= 10

    if r.ram >= ADV_THRESHOLDS["ram_critical"]:
        score -= 15
    elif r.ram >= ADV_THRESHOLDS["ram_warning"]:
        score -= 8

    if r.conn_out >= ADV_THRESHOLDS["conn_critical"]:
        score -= 10
    elif r.conn_out >= ADV_THRESHOLDS["conn_warning"]:
        score -= 5

    growth = r.encolados - prev_queue
    if growth >= ADV_THRESHOLDS["growth_critical"]:
        score -= 15
    elif growth >= ADV_THRESHOLDS["growth_warning"]:
        score -= 8

    if r.active_recips >= ADV_THRESHOLDS["active_recips_warning"]:
        score -= 7

    if r.error or str(r.status).lower() != "online":
        score = min(score, 20)

    return max(0, min(100, score))


def diagnose_node(r: Result, prev_queue: int) -> str:
    growth = r.encolados - prev_queue

    if r.error:
        return "Sin respuesta"
    if str(r.status).lower() != "online":
        return f"Status={r.status}"
    if r.encolados >= ADV_THRESHOLDS["stuck_min_queue"] and r.conn_out <= ADV_THRESHOLDS["smtp_low_conn_out"]:
        return "Queue sin salida"
    if growth >= ADV_THRESHOLDS["growth_critical"] and r.conn_out >= ADV_THRESHOLDS["smtp_high_conn_out"]:
        return "Queue creciendo"
    if r.cpu >= ADV_THRESHOLDS["cpu_critical"]:
        return "CPU alta"
    if r.ram >= ADV_THRESHOLDS["ram_critical"]:
        return "RAM alta"
    if r.conn_out >= ADV_THRESHOLDS["conn_critical"]:
        return "Salida alta"
    if r.active_recips >= ADV_THRESHOLDS["active_recips_warning"] and r.encolados >= 50:
        return "Muchos recips"
    if r.encolados == 0 and r.conn_in == 0 and r.conn_out == 0:
        return "Sin actividad"
    return "OK"


def natural_risk_text(cluster: str, nodo: str, diag: str) -> str:
    if diag == "Sin respuesta":
        return f"{RED}{nodo} de {cluster} sin respuesta{RESET}"

    mapping = {
        "Salida alta": f"{YELLOW}Salida alta en {nodo} ({cluster}){RESET}",
        "CPU alta": f"{YELLOW}CPU alta en {nodo} ({cluster}){RESET}",
        "RAM alta": f"{YELLOW}RAM alta en {nodo} ({cluster}){RESET}",
        "Queue sin salida": f"{YELLOW}Queue sin salida en {nodo} ({cluster}){RESET}",
        "Queue creciendo": f"{YELLOW}Cola creciendo en {nodo} ({cluster}){RESET}",
        "Muchos recips": f"{YELLOW}Muchos destinatarios en {nodo} ({cluster}){RESET}",
        "Sin actividad": f"{YELLOW}Sin actividad en {nodo} ({cluster}){RESET}",
    }

    if diag.startswith("Status="):
        status = diag.split("=", 1)[1]
        return f"{YELLOW}{nodo} de {cluster}: status {status}{RESET}"

    return mapping.get(diag, f"{YELLOW}{diag} en {nodo} ({cluster}){RESET}")


def cluster_rate_tendency(
    msg_in_1m: int,
    msg_in_5m: int,
    comp_1m: int,
    comp_5m: int,
    comp_15m: int,
    queue_delta: int,
    active_recips: int,
    bounce_1m: int,
) -> str:
    if queue_delta >= ADV_THRESHOLDS["growth_critical"]:
        return f"{RED}Cola creciendo{RESET}"
    if queue_delta <= -ADV_THRESHOLDS["growth_warning"]:
        return f"{GREEN}Cola drenando{RESET}"

    if active_recips >= ADV_THRESHOLDS["active_recips_warning"] * 2:
        if comp_1m < comp_5m:
            return f"{RED}Carga alta con proceso bajo{RESET}"
        return f"{YELLOW}Carga activa elevada{RESET}"

    if bounce_1m >= 1000:
        return f"{YELLOW}Rebotes temporales altos{RESET}"

    up_5 = comp_1m / max(comp_5m, 1)
    up_15 = comp_1m / max(comp_15m, 1)

    if up_5 >= 1.20 and up_15 >= 1.20:
        return f"{GREEN}Procesamiento acelerando{RESET}"

    if up_5 <= 0.80 and up_15 <= 0.80:
        return f"{RED}Procesamiento cayendo{RESET}"

    msg_ratio = msg_in_1m / max(msg_in_5m, 1)

    if msg_ratio >= 1.20 and queue_delta >= 0:
        return f"{YELLOW}Tráfico creciendo{RESET}"

    if msg_ratio <= 0.80 and queue_delta <= 0:
        return f"{GREEN}Entrada moderándose{RESET}"

    return "Estable"


def has_incidents(current_data: Dict[str, Dict[str, Result]]) -> bool:
    for cluster in cluster_order():
        for nodo in current_data.get(cluster, {}):
            r = current_data[cluster][nodo]

            if r.error:
                return True
            if str(r.status).lower() != "online":
                return True

            th = CLUSTER_THRESHOLDS[cluster]
            if r.encolados >= th["warning"]:
                return True
            if r.cpu >= ADV_THRESHOLDS["cpu_warning"]:
                return True
            if r.ram >= ADV_THRESHOLDS["ram_warning"]:
                return True
            if r.conn_out >= ADV_THRESHOLDS["conn_warning"]:
                return True

    return False

def update_cluster_peaks(
    current_data: Dict[str, Dict[str, Result]],
    node_peaks: Dict[str, Dict[str, Dict[str, object]]],
    total_peaks: Dict[str, Dict[str, object]],
    timestamp: str,
) -> None:
    for cluster, nodo_map in current_data.items():
        if cluster not in node_peaks:
            node_peaks[cluster] = {}

        if cluster not in total_peaks:
            total_peaks[cluster] = {"total": 0, "hora": "--:--"}

        total_cluster = 0

        for nodo, r in nodo_map.items():
            total_cluster += r.encolados

            prev_peak = node_peaks[cluster].get(nodo)
            if prev_peak is None or r.encolados > int(prev_peak["encolados"]):
                node_peaks[cluster][nodo] = {
                    "hora": timestamp,
                    "encolados": r.encolados,
                }

        if total_cluster > int(total_peaks[cluster]["total"]):
            total_peaks[cluster] = {
                "hora": timestamp,
                "total": total_cluster,
            }


def print_cluster_peaks_table(
    node_peaks: Dict[str, Dict[str, Dict[str, object]]],
    total_peaks: Dict[str, Dict[str, object]],
) -> None:
    print("PICOS DE ENCOLADOS POR CLUSTER\n")

    clusters = cluster_order()

    row_w = 9
    hora_w = 5
    nodo_w = 6
    enc_w = 9

    col_sep = " | "
    cluster_sep_plain = " || "
    cluster_sep_colored = f" {orange_cluster_sep()} "

    cluster_chunk_width = hora_w + len(col_sep) + nodo_w + len(col_sep) + enc_w

    h1 = f"| {'Nivel'.ljust(row_w)}"
    for cluster in clusters:
        h1 += f"{cluster_sep_colored}{cluster.center(cluster_chunk_width)}"
    h1 += " |"

    sub = (
        f"{'Hora'.center(hora_w)}{col_sep}"
        f"{'Nodo'.center(nodo_w)}{col_sep}"
        f"{'Encol.'.center(enc_w)}"
    )

    h2 = f"| {'':<{row_w}}"
    for _ in clusters:
        h2 += f"{cluster_sep_colored}{sub}"
    h2 += " |"

    sep = "-" * (
        2 + row_w +
        (len(cluster_sep_plain) + cluster_chunk_width) * len(clusters) + 2
    )

    print(h1)
    print(h2)
    print(sep)

    ranking_by_cluster: Dict[str, List[Tuple[str, str, int]]] = {}
    for cluster in clusters:
        all_nodes = []
        for nodo, info in node_peaks.get(cluster, {}).items():
            all_nodes.append((str(info["hora"]), nodo, int(info["encolados"])))
        all_nodes.sort(key=lambda x: x[2], reverse=True)
        ranking_by_cluster[cluster] = all_nodes

    for pos in range(3):
        line = f"| {('TOP ' + str(pos + 1)).ljust(row_w)}"
        for cluster in clusters:
            ranking = ranking_by_cluster.get(cluster, [])
            if pos < len(ranking):
                hora, nodo, encolados = ranking[pos]
                enc_txt = queue_color(cluster, fmt_num_dot(encolados).rjust(enc_w), encolados)
                chunk = f"{hora.ljust(hora_w)}{col_sep}{nodo.ljust(nodo_w)}{col_sep}{enc_txt}"
            else:
                chunk = f"{'--:--'.ljust(hora_w)}{col_sep}{'-'.ljust(nodo_w)}{col_sep}{'0'.rjust(enc_w)}"

            line += f"{cluster_sep_colored}{chunk}"

        line += " |"
        print(line)

    print(sep)

    total_line = f"| {'TOTAL MAX'.ljust(row_w)}"
    for cluster in clusters:
        info = total_peaks.get(cluster, {"hora": "--:--", "total": 0})
        hora = str(info["hora"])
        total = int(info["total"])
        total_txt = color_by_threshold(
            fmt_num_dot(total).rjust(enc_w),
            total,
            CLUSTER_THRESHOLDS[cluster]["total_warning"],
            CLUSTER_THRESHOLDS[cluster]["total_critical"],
        )
        chunk = f"{hora.ljust(hora_w)}{col_sep}{'TOTAL'.ljust(nodo_w)}{col_sep}{total_txt}"
        total_line += f"{cluster_sep_colored}{chunk}"

    total_line += " |"
    print(total_line)

def safe_cluster_filename(cluster: str) -> str:
    return (
        cluster.lower()
        .replace("á", "a")
        .replace("é", "e")
        .replace("í", "i")
        .replace("ó", "o")
        .replace("ú", "u")
        .replace(" ", "_")
    )


def ensure_csv_dir() -> None:
    os.makedirs(CSV_OUTPUT_DIR, exist_ok=True)


def build_cluster_csv_path(cluster: str, now: datetime) -> str:
    filename = f"{now.strftime('%Y_%m')}_{safe_cluster_filename(cluster)}.csv"
    return os.path.join(CSV_OUTPUT_DIR, filename)


def append_cluster_csv(cluster: str, results_map: Dict[str, Result], now: datetime) -> None:
    ensure_csv_dir()
    path = build_cluster_csv_path(cluster, now)
    file_exists = os.path.exists(path)

    with open(path, "a", newline="", encoding="utf-8") as f:
        writer = csv.writer(f, delimiter=";")
        if not file_exists:
            writer.writerow(CSV_HEADERS)

        for nodo in sorted(results_map.keys()):
            r = results_map[nodo]
            writer.writerow([
                now.strftime("%Y-%m-%d"),
                now.strftime("%H:%M:%S"),
                r.timestamp,
                r.cluster,
                r.nodo,
                r.status,
                r.encolados,
                r.cpu,
                r.ram,
                r.conn_in,
                r.conn_out,
                r.active_recips,
                r.inj_msgs_1m,
                r.inj_msgs_5m,
                r.inj_msgs_15m,
                r.delivered_recips_1m,
                r.delivered_recips_5m,
                r.delivered_recips_15m,
                r.inj_recips_1m,
                r.inj_recips_5m,
                r.inj_recips_15m,
                r.completed_recips_1m,
                r.completed_recips_5m,
                r.completed_recips_15m,
                r.soft_bounced_evts_1m,
                r.soft_bounced_evts_5m,
                r.soft_bounced_evts_15m,
                r.error or "",
            ])


def persist_current_data_to_csv(current_data: Dict[str, Dict[str, Result]], now: datetime) -> None:
    for cluster, results_map in current_data.items():
        append_cluster_csv(cluster, results_map, now)

# ESC LISTENER
# =========================================================
def esc_listener() -> None:
    if not WINDOWS:
        return
    while not stop_event.is_set():
        if msvcrt.kbhit():
            key = msvcrt.getch()
            if key == b"\x1b":
                stop_event.set()
                return
        time.sleep(0.1)

def parse_status(xml_text: str) -> Tuple[str, Dict[str, int]]:
    root = ET.fromstring(xml_text)

    system_status = "unknown"
    sys_node = root.find(".//system")
    if sys_node is not None:
        system_status = sys_node.attrib.get("status", system_status)

    gauges = {
        "msgs_in_work_queue": 0,
        "cpu_utilization": 0,
        "ram_utilization": 0,
        "conn_in": 0,
        "conn_out": 0,
        "active_recips": 0,
    }

    for gauge in root.findall(".//gauge"):
        name = gauge.attrib.get("name")
        if name in gauges:
            gauges[name] = to_int(gauge.attrib.get("current"))

    rate_names = [
        "inj_msgs",
        "inj_recips",
        "soft_bounced_evts",
        "completed_recips",
        "delivered_recips",
    ]

    rates: Dict[str, int] = {}
    for name in rate_names:
        rates[f"{name}_1m"] = 0
        rates[f"{name}_5m"] = 0
        rates[f"{name}_15m"] = 0

    for rate in root.findall(".//rate"):
        name = rate.attrib.get("name")
        if name in rate_names:
            rates[f"{name}_1m"] = to_int(rate.attrib.get("last_1_min"))
            rates[f"{name}_5m"] = to_int(rate.attrib.get("last_5_min"))
            rates[f"{name}_15m"] = to_int(rate.attrib.get("last_15_min"))

    return system_status, {**gauges, **rates}

def fetch_one(
    session: requests.Session,
    url: str,
    auth: Tuple[str, str],
    timestamp: str,
    cluster: str,
    idx: int,
) -> Result:
    nodo = nodo_from_index(idx)
    last_err: Optional[str] = None

    for _ in range(RETRIES + 1):
        try:
            r = session.get(
                url,
                auth=auth,
                timeout=TIMEOUT_SECONDS,
                verify=VERIFY_SSL,
                headers={"Accept": "application/xml"},
            )

            if r.status_code != 200:
                return Result(
                    timestamp=timestamp,
                    nodo=nodo,
                    cluster=cluster,
                    status="offline",
                    encolados=0,
                    cpu=0,
                    ram=0,
                    conn_in=0,
                    conn_out=0,
                    active_recips=0,
                    inj_msgs_1m=0,
                    inj_msgs_5m=0,
                    inj_msgs_15m=0,
                    delivered_recips_1m=0,
                    delivered_recips_5m=0,
                    delivered_recips_15m=0,
                    inj_recips_1m=0,
                    inj_recips_5m=0,
                    inj_recips_15m=0,
                    completed_recips_1m=0,
                    completed_recips_5m=0,
                    completed_recips_15m=0,
                    soft_bounced_evts_1m=0,
                    soft_bounced_evts_5m=0,
                    soft_bounced_evts_15m=0,
                    error=f"HTTP {r.status_code}",
                )

            status, m = parse_status(r.text)

            return Result(
                timestamp=timestamp,
                nodo=nodo,
                cluster=cluster,
                status=status,
                encolados=m["msgs_in_work_queue"],
                cpu=m["cpu_utilization"],
                ram=m["ram_utilization"],
                conn_in=m["conn_in"],
                conn_out=m["conn_out"],
                active_recips=m["active_recips"],
                inj_msgs_1m=m["inj_msgs_1m"],
                inj_msgs_5m=m["inj_msgs_5m"],
                inj_msgs_15m=m["inj_msgs_15m"],
                delivered_recips_1m=m["delivered_recips_1m"],
                delivered_recips_5m=m["delivered_recips_5m"],
                delivered_recips_15m=m["delivered_recips_15m"],
                inj_recips_1m=m["inj_recips_1m"],
                inj_recips_5m=m["inj_recips_5m"],
                inj_recips_15m=m["inj_recips_15m"],
                completed_recips_1m=m["completed_recips_1m"],
                completed_recips_5m=m["completed_recips_5m"],
                completed_recips_15m=m["completed_recips_15m"],
                soft_bounced_evts_1m=m["soft_bounced_evts_1m"],
                soft_bounced_evts_5m=m["soft_bounced_evts_5m"],
                soft_bounced_evts_15m=m["soft_bounced_evts_15m"],
                error=None,
            )

        except requests.exceptions.RequestException as e:
            last_err = str(e)
        except ET.ParseError as e:
            last_err = f"XML ParseError: {e}"

    return Result(
        timestamp=timestamp,
        nodo=nodo,
        cluster=cluster,
        status="offline",
        encolados=0,
        cpu=0,
        ram=0,
        conn_in=0,
        conn_out=0,
        active_recips=0,
        inj_msgs_1m=0,
        inj_msgs_5m=0,
        inj_msgs_15m=0,
        delivered_recips_1m=0,
        delivered_recips_5m=0,
        delivered_recips_15m=0,
        inj_recips_1m=0,
        inj_recips_5m=0,
        inj_recips_15m=0,
        completed_recips_1m=0,
        completed_recips_5m=0,
        completed_recips_15m=0,
        soft_bounced_evts_1m=0,
        soft_bounced_evts_5m=0,
        soft_bounced_evts_15m=0,
        error=last_err or "RequestException",
    )


def fetch_cluster(
    session: requests.Session,
    group: Dict,
    auth: Tuple[str, str],
    timestamp: str,
) -> List[Result]:
    cluster = group["cluster"]
    pattern = group["pattern"]
    start = int(group.get("start", 1))
    end = int(group.get("end", 20))

    results: List[Result] = []

    with ThreadPoolExecutor(max_workers=MAX_WORKERS) as ex:
        futures = [
            ex.submit(fetch_one, session, pattern.format(idx=i), auth, timestamp, cluster, i)
            for i in range(start, end + 1)
        ]
        for fut in as_completed(futures):
            results.append(fut.result())

    return sorted(results, key=lambda r: r.nodo)

def run_monitor(auth: Tuple[str, str]) -> None:
    previous_data: Dict[str, Dict[str, int]] = {}
    queue_history: Dict[str, Dict[str, List[int]]] = {}
    node_peaks: Dict[str, Dict[str, Dict[str, object]]] = {}
    total_peaks: Dict[str, Dict[str, object]] = {}

    listener = threading.Thread(target=esc_listener, daemon=True)
    listener.start()

    with requests.Session() as session:
        while not stop_event.is_set():
            now = datetime.now()
            timestamp = now.strftime("%H:%M")
            current_data: Dict[str, Dict[str, Result]] = {}

            with ThreadPoolExecutor(max_workers=len(GROUPS)) as ex:
                fut_map = {
                    ex.submit(fetch_cluster, session, g, auth, timestamp): g["cluster"]
                    for g in GROUPS
                }
                for fut in as_completed(fut_map):
                    cluster = fut_map[fut]
                    results = fut.result()
                    current_data[cluster] = {r.nodo: r for r in results}

            persist_current_data_to_csv(current_data, now)

            for cluster, nodo_map in current_data.items():
                queue_history.setdefault(cluster, {})
                for nodo, r in nodo_map.items():
                    queue_history[cluster].setdefault(nodo, []).append(r.encolados)
                    queue_history[cluster][nodo] = queue_history[cluster][nodo][-5:]

            update_cluster_peaks(current_data, node_peaks, total_peaks, timestamp)

            previous_data_snapshot = {
                c: {n: r.encolados for n, r in m.items()}
                for c, m in current_data.items()
            }

            incidents_active = has_incidents(current_data)
            refresh_seconds = REFRESH_SECONDS_INCIDENT if incidents_active else REFRESH_SECONDS_NORMAL
            next_refresh_time = datetime.now() + timedelta(seconds=refresh_seconds)

            def draw() -> None:
                print_header_with_timer("MONITOR COLAS ESA", next_refresh_time, refresh_seconds)

                if incidents_active:
                    print(f"{YELLOW}Modo incidencias activo{RESET}\n")
                else:
                    print(f"{GREEN}Modo normal{RESET}\n")

                print_summary_alerts_risk(current_data, previous_data)
                print()
                print_orange_separator()
                print()

                print_queue_table(timestamp, current_data, previous_data)
                print()
                print_orange_separator()
                print()

                print_cluster_peaks_table(node_peaks, total_peaks)
                print()
                print_orange_separator()
                print()

                print_node_health_table(current_data, previous_data)
                print()
                print_orange_separator()
                print()

                print_rates_table(current_data, previous_data)
                print()
                print_orange_separator()
                print()

                print_status_incidents(current_data)
                print()
                print_orange_separator()
                print()

                print_advanced_findings(current_data, previous_data, queue_history)

                print(f"\nCSV mensuales en: {CSV_OUTPUT_DIR}")
                if WINDOWS:
                    print("Pulsa ESC para salir")
                else:
                    print("Pulsa Ctrl+C para salir")

            render_screen(draw)

            slept = 0.0
            while slept < refresh_seconds and not stop_event.is_set():
                time.sleep(0.1)
                slept += 0.1

            previous_data = previous_data_snapshot

    print("Monitor detenido")

def main() -> None:
    if not os.path.exists(CREDENTIALS_FILE):
        raise FileNotFoundError(
            f"No existe el archivo '{CREDENTIALS_FILE}'. Debe contener GLOBAL_USERNAME y GLOBAL_PASSWORD"
        )

    creds = load_env_file(CREDENTIALS_FILE)

    if "GLOBAL_USERNAME" not in creds or "GLOBAL_PASSWORD" not in creds:
        raise KeyError("Faltan GLOBAL_USERNAME o GLOBAL_PASSWORD en credentials.env")

    auth = (creds["GLOBAL_USERNAME"], creds["GLOBAL_PASSWORD"])

    ensure_csv_dir()
    run_monitor(auth)


if __name__ == "__main__":
    try:
        main()
    except KeyboardInterrupt:
        stop_event.set()
        print("\nMonitor detenido por usuario")
