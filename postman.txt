#!/usr/bin/env python3
# -*- coding: utf-8 -*-

import os
import csv
from dataclasses import dataclass
from typing import Optional, List, Tuple, Dict
from datetime import datetime

import requests
import urllib3
import xml.etree.ElementTree as ET
from concurrent.futures import ThreadPoolExecutor, as_completed


# -------------------------
# Config
# -------------------------
CREDENTIALS_FILE = "credentials.env"
OUTPUT_FOLDER = "encolados"

SERVER_PATTERN = "https://dh529-esa{idx}.c3s2.iphmx.com"

START_IDX = 1
END_IDX = 20

TIMEOUT_SECONDS = 12
MAX_WORKERS = 10
RETRIES = 2

VERIFY_SSL = False

if not VERIFY_SSL:
    urllib3.disable_warnings(urllib3.exceptions.InsecureRequestWarning)


@dataclass
class Result:
    timestamp: str
    nodo: str
    cluster: str
    status: str
    gauge_name: str
    encolados: int
    error: Optional[str] = None


def script_dir():
    return os.path.dirname(os.path.abspath(__file__))


def load_env_file(path: str):

    data = {}

    with open(path, "r") as f:

        for line in f:

            line = line.strip()

            if not line or line.startswith("#"):
                continue

            if "=" not in line:
                continue

            k, v = line.split("=", 1)

            data[k.strip()] = v.strip()

    return data


def nodo_from_index(idx: int):

    return f"ESA{idx:02d}"


def get_cluster_from_url(url: str):

    if "dh529" in url.lower():
        return "Europa"

    return "Otro"


def parse_encolados(xml_text):

    root = ET.fromstring(xml_text)

    system_status = "No disponible"

    sys_node = root.find(".//system")

    if sys_node is not None:
        system_status = sys_node.attrib.get("status", system_status)

    for gauge in root.findall(".//gauge"):

        if gauge.attrib.get("name") == "msgs_in_work_queue":

            val = gauge.attrib.get("current")

            if val is None:

                current_node = gauge.find("./current")

                if current_node is not None and current_node.text:
                    val = current_node.text.strip()

            if val is None or val.upper() == "N/A":
                return system_status, 0

            try:
                return system_status, int(float(val))
            except:
                return system_status, 0

    return system_status, 0


def fetch_one(session, base_url, auth, idx, timestamp):

    nodo = nodo_from_index(idx)

    cluster = get_cluster_from_url(base_url)

    url = f"{base_url}/xml/status"

    last_err = None

    for _ in range(RETRIES + 1):

        try:

            r = session.get(
                url,
                headers={"Accept": "application/xml"},
                auth=auth,
                timeout=TIMEOUT_SECONDS,
                verify=VERIFY_SSL
            )

            if r.status_code != 200:

                return Result(
                    timestamp,
                    nodo,
                    cluster,
                    "Error conexión",
                    "msgs_in_work_queue",
                    0,
                    f"HTTP {r.status_code}"
                )

            status, encolados = parse_encolados(r.text)

            return Result(
                timestamp,
                nodo,
                cluster,
                status,
                "msgs_in_work_queue",
                encolados
            )

        except requests.exceptions.RequestException as e:

            last_err = str(e)

    return Result(
        timestamp,
        nodo,
        cluster,
        "Error conexión",
        "msgs_in_work_queue",
        0,
        last_err
    )


def print_table(results):

    headers = ["Timestamp", "Nodo", "Cluster", "Status", "Gauge", "Encolados", "Error"]

    rows = []

    for r in results:

        rows.append([
            r.timestamp,
            r.nodo,
            r.cluster,
            r.status,
            r.gauge_name,
            r.encolados,
            r.error or ""
        ])

    widths = [len(h) for h in headers]

    for row in rows:
        for i, c in enumerate(row):
            widths[i] = max(widths[i], len(str(c)))

    def fmt(vals):
        return " | ".join(str(v).ljust(widths[i]) for i, v in enumerate(vals))

    sep = "-+-".join("-" * w for w in widths)

    print(fmt(headers))
    print(sep)

    for r in rows:
        print(fmt(r))


def save_csv(results, path):

    with open(path, "w", newline="", encoding="utf-8") as f:

        writer = csv.writer(f)

        writer.writerow([
            "timestamp",
            "nodo",
            "cluster",
            "status",
            "gauge",
            "encolados",
            "error"
        ])

        for r in results:

            writer.writerow([
                r.timestamp,
                r.nodo,
                r.cluster,
                r.status,
                r.gauge_name,
                r.encolados,
                r.error or ""
            ])


def main():

    now = datetime.now()

    ts_column = now.strftime("%Y-%m-%d %H:%M:%S")

    ts_file = now.strftime("%Y%m%d_%H%M%S")

    base_path = script_dir()

    # Crear carpeta encolados si no existe
    output_path = os.path.join(base_path, OUTPUT_FOLDER)
    os.makedirs(output_path, exist_ok=True)

    creds = load_env_file(os.path.join(base_path, CREDENTIALS_FILE))

    username = creds["GLOBAL_USERNAME"]
    password = creds["GLOBAL_PASSWORD"]

    servers = []

    for i in range(START_IDX, END_IDX + 1):

        servers.append((i, SERVER_PATTERN.format(idx=i)))

    results = []

    with requests.Session() as session:

        with ThreadPoolExecutor(max_workers=MAX_WORKERS) as ex:

            futures = []

            for idx, url in servers:

                futures.append(
                    ex.submit(fetch_one, session, url, (username, password), idx, ts_column)
                )

            for f in as_completed(futures):

                results.append(f.result())

    results.sort(key=lambda r: r.encolados, reverse=True)

    print_table(results)

    csv_name = f"encolados_europa_{ts_file}.csv"

    csv_path = os.path.join(output_path, csv_name)

    save_csv(results, csv_path)

    print("\nCSV guardado en:", csv_path)


if __name__ == "__main__":
    main()
