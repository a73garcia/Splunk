#!/usr/bin/env python3
# -*- coding: utf-8 -*-

"""
Monitor Cisco ESA (/xml/status) - Encolados por cluster

✅ Lee credenciales desde credentials.env (misma carpeta)
✅ Consulta en paralelo un rango de nodos por cluster
✅ Soporta certificados self-signed (verify=False) y silencia warnings
✅ Extrae:
   - status/system/@status
   - gauge name="msgs_in_work_queue" (valor viene como atributo current="17")
✅ Resultados:
   - Nodo mostrado: ESA01..ESA20 (con 0 delante solo en resultados)
   - Cluster: Europa, Aplicaciones, América, Brasil
✅ Encolados: N/A o ausente => 0
✅ Consola:
   - Tabla completa por cluster (ordenada por encolados desc)
   - Tabla resumen final por cluster (total, top, errores)
✅ Persistencia:
   - Carpeta encolados/
   - 1 CSV por cluster y por mes (append): encolados_<Cluster>_YYYY-MM.csv
"""

import os
import csv
from dataclasses import dataclass
from typing import Optional, List, Dict, Tuple
from datetime import datetime

import requests
import urllib3
import xml.etree.ElementTree as ET
from concurrent.futures import ThreadPoolExecutor, as_completed


# -------------------------
# Config
# -------------------------
CREDENTIALS_FILE = "credentials.env"
OUTPUT_FOLDER = "encolados"

TIMEOUT_SECONDS = 12
MAX_WORKERS = 12
RETRIES = 2

# SSL (ESA suele usar self-signed)
VERIFY_SSL = False
if not VERIFY_SSL:
    urllib3.disable_warnings(urllib3.exceptions.InsecureRequestWarning)

# Un CSV mensual por cluster (append)
# idx en URL SIN 0 delante: ...esa1..., ...esa2...
# Nodo mostrado CON 0 delante: ESA01, ESA02...
GROUPS = [
    {
        "cluster": "Europa",
        "pattern": "https://dh529-esa{idx}.c3s2.iphmx.com/xml/status",
        "start": 1,
        "end": 20,
    },
    {
        "cluster": "Aplicaciones",
        "pattern": "https://dh5505-esa{idx}.c3s2.iphmx.com/xml/status",
        "start": 1,
        "end": 20,
    },
    {
        "cluster": "América",
        "pattern": "https://dh5504-esa{idx}.c3s2.iphmx.com/xml/status",
        "start": 1,
        "end": 20,
    },
    {
        "cluster": "Brasil",
        "pattern": "https://dh6121-esa{idx}.iphmx.com/xml/status",
        "start": 1,
        "end": 20,
    },
]


# -------------------------
# Data model
# -------------------------
@dataclass
class Result:
    timestamp: str
    nodo: str
    cluster: str
    status: str
    gauge: str
    encolados: int
    error: Optional[str] = None


# -------------------------
# Helpers
# -------------------------
def script_dir() -> str:
    return os.path.dirname(os.path.abspath(__file__))


def load_env_file(path: str) -> Dict[str, str]:
    data: Dict[str, str] = {}
    if not os.path.exists(path):
        raise FileNotFoundError(f"No existe el archivo de credenciales: {path}")

    with open(path, "r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line or line.startswith("#") or "=" not in line:
                continue
            k, v = line.split("=", 1)
            data[k.strip()] = v.strip()
    return data


def nodo_from_index(idx: int) -> str:
    return f"ESA{idx:02d}"


def safe_filename(name: str) -> str:
    return (
        name.replace(" ", "_")
        .replace("/", "_")
        .replace("\\", "_")
        .replace(":", "_")
        .replace("á", "a")
        .replace("é", "e")
        .replace("í", "i")
        .replace("ó", "o")
        .replace("ú", "u")
        .replace("Á", "A")
        .replace("É", "E")
        .replace("Í", "I")
        .replace("Ó", "O")
        .replace("Ú", "U")
    )


def monthly_csv_path(out_dir: str, cluster: str, now: datetime) -> str:
    month = now.strftime("%Y-%m")
    return os.path.join(out_dir, f"encolados_{safe_filename(cluster)}_{month}.csv")


# -------------------------
# XML parsing
# -------------------------
def parse_encolados(xml_text: str) -> Tuple[str, int]:
    """
    ESA típico:
      <gauge name="msgs_in_work_queue" current="17"/>
    Si falta o es N/A => 0
    """
    root = ET.fromstring(xml_text)

    system_status = "No disponible"
    sys_node = root.find(".//system")
    if sys_node is not None:
        system_status = sys_node.attrib.get("status", system_status)

    for gauge in root.findall(".//gauge"):
        if gauge.attrib.get("name") == "msgs_in_work_queue":
            val = gauge.attrib.get("current")

            # Fallback por si viniera como nodo <current>17</current>
            if val is None:
                current_node = gauge.find("./current")
                if current_node is not None and current_node.text:
                    val = current_node.text.strip()

            if val is None or val.strip().upper() == "N/A":
                return system_status, 0

            try:
                return system_status, int(float(val.strip()))
            except Exception:
                return system_status, 0

    return system_status, 0


# -------------------------
# Network
# -------------------------
def fetch_one(
    session: requests.Session,
    url: str,
    auth: Tuple[str, str],
    timestamp: str,
    cluster: str,
    idx: int,
) -> Result:
    node = nodo_from_index(idx)
    headers = {"Accept": "application/xml"}

    last_err: Optional[str] = None
    for _ in range(RETRIES + 1):
        try:
            r = session.get(
                url,
                headers=headers,
                auth=auth,
                timeout=TIMEOUT_SECONDS,
                verify=VERIFY_SSL,
            )

            if r.status_code != 200:
                return Result(
                    timestamp=timestamp,
                    nodo=node,
                    cluster=cluster,
                    status="Error conexión",
                    gauge="msgs_in_work_queue",
                    encolados=0,
                    error=f"HTTP {r.status_code}",
                )

            status, q = parse_encolados(r.text)
            return Result(
                timestamp=timestamp,
                nodo=node,
                cluster=cluster,
                status=status,
                gauge="msgs_in_work_queue",
                encolados=q,
                error=None,
            )

        except requests.exceptions.RequestException as e:
            last_err = str(e)

    return Result(
        timestamp=timestamp,
        nodo=node,
        cluster=cluster,
        status="Error conexión",
        gauge="msgs_in_work_queue",
        encolados=0,
        error=last_err or "RequestException",
    )


# -------------------------
# Output: CSV (append mensual)
# -------------------------
def append_csv(results: List[Result], path: str) -> None:
    fieldnames = ["timestamp", "nodo", "cluster", "status", "gauge", "encolados", "error"]
    file_exists = os.path.exists(path)

    with open(path, "a", newline="", encoding="utf-8") as f:
        w = csv.DictWriter(f, fieldnames=fieldnames)
        if not file_exists:
            w.writeheader()

        for r in results:
            w.writerow(
                {
                    "timestamp": r.timestamp,
                    "nodo": r.nodo,
                    "cluster": r.cluster,
                    "status": r.status,
                    "gauge": r.gauge,
                    "encolados": r.encolados,
                    "error": r.error or "",
                }
            )


# -------------------------
# Output: Console tables
# -------------------------
def print_table(cluster: str, results: List[Result]) -> None:
    results_sorted = sorted(results, key=lambda r: r.encolados, reverse=True)

    headers = ["Timestamp", "Nodo", "Cluster", "Status", "Gauge", "Encolados", "Error"]
    rows = [
        [r.timestamp, r.nodo, r.cluster, r.status, r.gauge, str(r.encolados), (r.error or "")]
        for r in results_sorted
    ]

    widths = [len(h) for h in headers]
    for row in rows:
        for i, cell in enumerate(row):
            widths[i] = max(widths[i], len(str(cell)))

    def fmt(vals):
        return " | ".join(str(v).ljust(widths[i]) for i, v in enumerate(vals))

    sep = "-+-".join("-" * w for w in widths)

    print(f"\n=== {cluster} ===")
    print(fmt(headers))
    print(sep)
    for row in rows:
        print(fmt(row))


def print_summary(all_results: Dict[str, List[Result]]) -> None:
    print("\n============== RESUMEN CLUSTERS ==============")

    headers = ["Cluster", "Total Encolados", "Nodo Máximo", "Top 3 Nodos", "Errores"]

    rows: List[List[str]] = []
    for cluster, results in all_results.items():
        total = sum(r.encolados for r in results)
        sorted_nodes = sorted(results, key=lambda r: r.encolados, reverse=True)

        max_node = sorted_nodes[0].nodo if sorted_nodes else "-"
        top3 = ", ".join([f"{r.nodo}:{r.encolados}" for r in sorted_nodes[:3]]) if sorted_nodes else "-"
        errors = sum(1 for r in results if r.error)

        rows.append([cluster, str(total), max_node, top3, str(errors)])

    widths = [len(h) for h in headers]
    for row in rows:
        for i, cell in enumerate(row):
            widths[i] = max(widths[i], len(cell))

    def fmt(vals):
        return " | ".join(vals[i].ljust(widths[i]) for i in range(len(vals)))

    sep = "-+-".join("-" * w for w in widths)

    print(fmt(headers))
    print(sep)
    for row in rows:
        print(fmt(row))


# -------------------------
# Main
# -------------------------
def main() -> int:
    now = datetime.now()
    ts_col = now.strftime("%Y-%m-%d %H:%M:%S")

    base = script_dir()
    out_dir = os.path.join(base, OUTPUT_FOLDER)
    os.makedirs(out_dir, exist_ok=True)

    creds = load_env_file(os.path.join(base, CREDENTIALS_FILE))
    username = creds.get("GLOBAL_USERNAME")
    password = creds.get("GLOBAL_PASSWORD")
    if not username or not password:
        print("Faltan GLOBAL_USERNAME y/o GLOBAL_PASSWORD en credentials.env")
        return 2

    auth = (username, password)

    all_results: Dict[str, List[Result]] = {}

    with requests.Session() as session:
        for g in GROUPS:
            cluster = g["cluster"]
            pattern = g["pattern"]
            start = int(g["start"])
            end = int(g["end"])

            targets = [(i, pattern.format(idx=i)) for i in range(start, end + 1)]

            results: List[Result] = []
            with ThreadPoolExecutor(max_workers=MAX_WORKERS) as ex:
                futures = [
                    ex.submit(fetch_one, session, url, auth, ts_col, cluster, idx)
                    for idx, url in targets
                ]
                for fut in as_completed(futures):
                    results.append(fut.result())

            all_results[cluster] = results

            # Tabla completa por cluster
            print_table(cluster, results)

            # Append a CSV mensual del cluster
            csv_path = monthly_csv_path(out_dir, cluster, now)
            append_csv(results, csv_path)
            print(f"\n[{cluster}] Filas añadidas: {len(results)} | CSV mensual: {csv_path}")

    # Resumen final
    print_summary(all_results)

    return 0


if __name__ == "__main__":
    raise SystemExit(main())
